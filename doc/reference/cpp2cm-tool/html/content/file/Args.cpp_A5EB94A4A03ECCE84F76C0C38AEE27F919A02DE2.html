11 {
12     std::vector<int> argsVec;
13     TrivialLexer lexer(args, "", 0);

15     return argsVec;

18 int ParseInt(const std::u32string& s)
19 {
20     int x = 0;
21     for (char32_t c : s)
22     {
23         if (c >= '0' && c <= '9')
24         {
25             x = 10 * x + c - '0';
26         }
27     }
28     return x;
29 }
