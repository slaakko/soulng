 55 std::u32string TypeName::ToString() const
 56 {
 57     std::u32string s(Name());
 58     if (isTemplate)
 59     {
 60         s.append(U"<");
 61         int n = int(templateArguments.size());
 62         for (int i = 0; i < n; ++i)
 63         {
 64             if (i > 0)
 65             {
 66                 s.append(U", ");
 67             }
 68             s.append(templateArguments[i]->ToString());
 69         }
 70         s.append(U">");
 71     }
 72     return s;
 73 }

130 std::u32string TypeId::ToString() const
131 {
132     std::u32string v;
133     int n = int(typeSpecifiers.size());
134     for (int i = 0; i < n; ++i)
135     {
136         if (i > 0)
137         {
138             v.append(U" ");
139         }
140         v.append(typeSpecifiers[i]->ToString());
141     }
142     if (!declarator.empty())
143     {
144         v.append(declarator);
145     }
146     return v;
147 }

149 void TypeId::Print(CodeFormatter& formatter)
150 {
151     int n = int(typeSpecifiers.size());
152     for (int i = 0; i < n; ++i)
153     {
154         if (i > 0)
155         {
156             formatter.Write(" ");
157         }
158         typeSpecifiers[i]->Print(formatter);
159     }
160     if (!declarator.empty())
161     {
162         formatter.Write(ToUtf8(declarator));
163     }
164 }

166 void TypeId::PrintNonPtrType(CodeFormatter& formatter)
167 {
168     if (IsPtrType())
169     {
170         int n = int(typeSpecifiers.size());
171         for (int i = 0; i < n; ++i)
172         {
173             if (i > 0)
174             {
175                 formatter.Write(" ");
176             }
177             typeSpecifiers[i]->Print(formatter);
178         }
179         if (!declarator.empty())
180         {
181             formatter.Write(ToUtf8(declarator.substr(0, declarator.size() - 1)));
182         }
183     }
184     else
185     {
186         Print(formatter);
187     }
188 }
