125 Operator OperatorTable::GetOperator(const std::u32string& operatorName)
126 {
127     std::unordered_map<Operator, Operator>                       i = operatorMap.find(operatorName);
128     if (i != operatorMap.end())
129     {
130         return i->second;
131     }
132     throw std::runtime_error("unknown operator '" + ToUtf8(operatorName) + "'");
133 }

135 std::u32string OperatorTable::GetOperatorStr(Operator op)
136 {
137     std::unordered_map<Operator, OperatorHash, OperatorHash>                   i = operatorStrMap.find(op);
138     if (i != operatorStrMap.end())
139     {
140         return i->second;
141     }
142     throw std::runtime_error("unknown operator");
143 }
