 28 void AssignInit::Print(CodeFormatter& formatter)
 29 {
 30     if (assignmentExpr)
 31     {
 32         assignmentExpr->Print(formatter);
 33     }
 34     else
 35     {
 36         formatter.Write("{ ");
 37         int n = int(subInits.size());
 38         for (int i = 0; i < n; ++i)
 39         {
 40             if (i > 0)
 41             {
 42                 formatter.Write(", ");
 43             }
 44             subInits[i]->Print(formatter);
 45         }
 46         formatter.Write(" }");
 47     }
 48 }

 63         }
 64     }
 65 }

 78 void Initializer::Print(CodeFormatter& formatter)
 79 {
 80     if (assignInit)
 81     {
 82         formatter.Write(" = ");
 83         assignInit->Print(formatter);
 84     }
 85     else if (!expressionList.empty())
 86     {
 87         formatter.Write("(");
 88         PrintExprs(expressionList, formatter);
 89         formatter.Write(")");
 90     }
 91 }

 93 void Initializer::Accept(Visitor& visitor)
 94 {
 95     visitor.Visit(*this);
 96     if (assignInit)
 97     {
 98         assignInit->Accept(visitor);
 99     }
100     else if (!expressionList.empty())
101     {
102         for (CppObject* expr : expressionList)
103         {
104             expr->Accept(visitor);
105         }
106     }
