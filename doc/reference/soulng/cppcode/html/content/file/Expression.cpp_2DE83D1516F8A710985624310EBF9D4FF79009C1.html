 16 void PrintExprs(const std::std::vector<CppObject*>&expressions, CodeFormatter&formatter)
 17 {
 18     bool first = true;
 19     for (CppObject* expr : expressions)
 20     {
 21         if (first)
 22         {
 23             first = false;
 24         }
 25         else
 26         {
 27             formatter.Write(", ");
 28         }
 29         expr->Print(formatter);
 30     }
 31 }

431 void NewExpr::Print(CodeFormatter& formatter)
432 {
433     if (global)
434     {
435         formatter.Write("::");
436     }
437     formatter.Write("new");
438     if (!placement.empty())
439     {
440         formatter.Write("(");
441         PrintExprs(placement, formatter);
442         formatter.Write(")");
443     }
444     if (parens)
445     {
446         formatter.Write("(");
447     }
448     formatter.Write(" ");
449     typeId->Print(formatter);
450     if (parens)
451     {
452         formatter.Write(")");
453     }
454     if (!initializer.empty())
455     {
456         formatter.Write("(");
457         PrintExprs(initializer, formatter);
458         formatter.Write(")");
459     }
460 }
461 
462 void NewExpr::Accept(Visitor& visitor)
463 {
464     visitor.Visit(*this);
465     for (CppObject* p : placement)
466     {
467         p->Accept(visitor);
468     }
469     if (typeId)
470     {
471         typeId->Accept(visitor);
472     }
473     for (CppObject* i : initializer)
474     {
475         i->Accept(visitor);
476     }
477 }
