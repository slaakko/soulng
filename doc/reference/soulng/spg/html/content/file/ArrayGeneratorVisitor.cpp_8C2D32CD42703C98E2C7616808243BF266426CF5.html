 12 ArrayGeneratorVisitor::ArrayGeneratorVisitor(CodeFormatter* formatter_, int& sn_) : formatter(formatter_), sn(sn_)
 13 {
 14 }
 15 
 16 void ArrayGeneratorVisitor::Visit(StringParser& parser)
 17 {
 18     std::string s = "s" + std::to_string(sn++);
 19     parser.SetArrayName(s);
 20     formatter->Write("static const int " + s + "[] = {");
 21     bool first = true;
 22     for (char32_t c : parser.Str())
 23     {
 24         if (first)
 25         {
 26             first = false;
 27         }
 28         else
 29         {
 30             formatter->Write(", ");
 31         }
 32         formatter->Write(std::to_string(static_cast<int>(c)));
 33     }
 34     formatter->WriteLine("};");
 35     formatter->WriteLine();
 36 }
 37 
 38 void ArrayGeneratorVisitor::Visit(CharSetParser& parser)
 39 {
 40     std::string s = "s" + std::to_string(sn++);
 41     parser.SetArrayName(s);
 42     formatter->Write("static const soulng::parser::Range " + s + "[] = {");
 43     bool first = true;
 44     for (const soulng::parser::Range& range : parser.Set().Ranges())
 45     {
 46         if (first)
 47         {
 48             first = false;
 49         }
 50         else
 51         {
 52             formatter->Write(", ");
 53         }
 54         formatter->Write("{" + std::to_string(range.first) + ", " + std::to_string(range.last) + "}");
 55     }
 56     formatter->WriteLine("};");
 57     formatter->WriteLine();
 58 }
