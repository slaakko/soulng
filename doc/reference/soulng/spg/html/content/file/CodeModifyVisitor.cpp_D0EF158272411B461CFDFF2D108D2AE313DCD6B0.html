20 void CodeModifyVisitor::Visit(soulng::cppcode::ReturnStatement& object)
21 {
22     std::vector<soulng::cppcode::CppObject*> matchArgs;
23     matchArgs.push_back(new soulng::cppcode::Literal(U"true"));
24     soulng::cppcode::CppObject* value = object.GetExpression();
25     if (!valueOfPtrType)
26     {
27         soulng::cppcode::TypeId* typeId = new soulng::cppcode::TypeId();
28         typeId->Add(new soulng::cppcode::TypeSpecifier(U"soulng::parser::Value<" + returnType->ToString() + U">"));
29         soulng::cppcode::NewExpr* newExpr = new soulng::cppcode::NewExpr(false);
30         newExpr->SetTypeId(typeId);
31         newExpr->BeginAddInitializer();
32         newExpr->AddExpr(value);
33         newExpr->EndAddInitializer();
34         value = newExpr;
35     }
36     matchArgs.push_back(value);
37     soulng::cppcode::InvokeExpr* invokeExpr = new soulng::cppcode::InvokeExpr(new soulng::cppcode::IdExpr(U"soulng::parser::Match"));
38     for (const auto& arg : matchArgs)
39     {
40         invokeExpr->AddExpr(arg);
41     }
42     object.SetExpression(invokeExpr);
43     if (!noParserDebugSupport)
44     {
45         soulng::cppcode::Statement* parent = object.Parent();
46         soulng::cppcode::CompoundStatement* debugBlock = new soulng::cppcode::CompoundStatement();
47         debugBlock->Add(new soulng::cppcode::IfDefStatement(new soulng::cppcode::Literal(U"SOULNG_PARSER_DEBUG_SUPPORT")));
48         soulng::cppcode::InvokeExpr* writeSuccessToLogCall = new soulng::cppcode::InvokeExpr(new soulng::cppcode::IdExpr(U"soulng::lexer::WriteSuccessToLog"));
49         writeSuccessToLogCall->AddExpr(new soulng::cppcode::IdExpr(U"lexer"));
50         writeSuccessToLogCall->AddExpr(new soulng::cppcode::IdExpr(U"parser_debug_match_span"));
51         soulng::cppcode::InvokeExpr* parserNameToUtf32 = new soulng::cppcode::InvokeExpr(new soulng::cppcode::IdExpr(U"soulng::unicode::ToUtf32"));
52         parserNameToUtf32->AddExpr(new soulng::cppcode::Literal(U"\"" + parserName + U"\""));
53         writeSuccessToLogCall->AddExpr(parserNameToUtf32);
54         soulng::cppcode::ExpressionStatement* writeSuccessToLogCallStatement = new soulng::cppcode::ExpressionStatement(writeSuccessToLogCall);
55         soulng::cppcode::IfStatement* ifS = new soulng::cppcode::IfStatement(new soulng::cppcode::Literal(U"parser_debug_write_to_log"), writeSuccessToLogCallStatement, nullptr);
56         debugBlock->Add(ifS);
57         debugBlock->Add(new soulng::cppcode::EndIfStatement(new soulng::cppcode::Literal(U"// SOULNG_PARSER_DEBUG_SUPPORT")));
58         debugBlock->Add(&object);
59         parent->Replace(&object, debugBlock);
60     }
61 }

63 void CodeModifyVisitor::Visit(soulng::cppcode::IdExpr& object)
64 {
65     for (auto& info : nonterminalInfos)
66     {
67         if (object.Name() == info.name)
68         {
69             if (info.ptrType)
70             {
71                 object.SetName(info.name + U".release()");
72             }
73             else
74             {
75                 object.SetName(info.name + U"->value");
76             }
77         }
78     }
79 }
