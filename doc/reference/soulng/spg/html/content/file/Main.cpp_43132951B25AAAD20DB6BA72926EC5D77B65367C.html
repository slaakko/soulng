 21 void PrintHelp()
 22 {
 23     std::cout << "Soul NG Parser Generator version " << soulng::spg::ParserGeneratorVersionStr() << std::endl;
 24     std::cout << "Usage: spg [options] <file.spg>" << std::endl;
 25     std::cout << "Options:" << std::endl;
 26     std::cout << "--help | -h" << std::endl;
 27     std::cout << "  Print help and exit." << std::endl;
 28     std::cout << "--verbose | -v" << std::endl;
 29     std::cout << "  Be verbose." << std::endl;
 30     std::cout << "--no-parser-debug-support | -n" << std::endl;
 31     std::cout << "  Do not generate parser debug support code to the generated parsers." << std::endl;
 32 }
 33 
 34 using namespace soulng::unicode;
 35 using namespace soulng::util;
 36 
 37 struct Initializer 
 38 {
 39     Initializer()
 40     {
 41         soulng::util::Init();
 42         soulng::cppcode::Init();
 43     }
 44     ~Initializer()
 45     {
 46         soulng::cppcode::Done();
 47         soulng::util::Done();
 48     }
 49 };
 50 
 51 int main(int argc, const char** argv)
 52 {
 53     Initializer initializer;
 54     try
 55     {
 56         bool verbose = false;
 57         bool noParserDebugSupport = false;
 58         std::string projectFilePath;
 59         for (int i = 1; i < argc; ++i)
 60         {
 61             std::string arg = argv[i];
 62             if (StartsWith(arg, "--"))
 63             {
 64                 if (arg == "--help")
 65                 {
 66                     PrintHelp();
 67                     return 1;
 68                 }
 69                 else if (arg == "--verbose")
 70                 {
 71                     verbose = true;
 72                 }
 73                 else if (arg == "--no-parser-debug-support")
 74                 {
 75                     noParserDebugSupport = true;
 76                 }
 77                 else
 78                 {
 79                     throw std::runtime_error("unknown option '" + arg + "'");
 80                 }
 81             }
 82             else if (StartsWith(arg, "-"))
 83             {
 84                 std::string options = arg.substr(1);
 85                 if (options.empty())
 86                 {
 87                     throw std::runtime_error("unknown option '" + arg + "'");
 88                 }
 89                 else
 90                 {
 91                     for (char o : options)
 92                     {
 93                         if (o == 'h')
 94                         {
 95                             PrintHelp();
 96                             return 1;
 97                         }
 98                         else if (o == 'v')
 99                         {
100                             verbose = true;
101                         }
102                         else if (o == 'n')
103                         {
104                             noParserDebugSupport = true;
105                         }
106                         else
107                         {
108                             throw std::runtime_error("unknown option '-" + std::string(1, o) + "'");
109                         }
110                     }
111                 }
112             }
113             else
114             {
115                 std::string filePath = GetFullPath(arg);
116                 if (Path::GetExtension(filePath) != ".spg")
117                 {
118                     throw std::runtime_error("unknown project file extension '" + Path::GetExtension(filePath) + "'");
119                 }
120                 projectFilePath = filePath;
121             }
122         }
123         if (projectFilePath.empty())
124         {
125             PrintHelp();
126             return 1;
127         }
128         if (verbose)
129         {
130             std::cout << "> " << projectFilePath << std::endl;
131         }
132         std::string s = ReadFile(projectFilePath);
133         std::u32string content = ToUtf32(s);
134         ProjectFileLexer lexer(content, projectFilePath, 0);
135         std::unique_ptr<soulng::spg::ProjectFile> projectFile = ProjectFileParser::Parse(lexer);
136         int index = 1;
137         std::vector<std::vector<std::ParserFile>>             parserFiles;
138         for (const std::string& sourceFileName : projectFile->SourceFiles())
139         {
140             std::string sourceFilePath = Path::Combine(projectFile->BasePath(), sourceFileName);
141             std::string s = ReadFile(sourceFilePath);
142             std::u32string content = ToUtf32(s);
143             ParserFileLexer lexer(content, sourceFilePath, index);
144             std::unique_ptr<soulng::spg::ParserFile> parserFile = ParserFileParser::Parse(lexer);
145             ++index;
146             parserFiles.push_back(std::move(parserFile));
147         }
148         soulng::spg::Domain domain;
149         for (const auto& parserFile : parserFiles)
150         {
151             domain.AddParserFile(parserFile.get());
152         }
153         soulng::spg::LinkerVisitor linkerVisitor;
154         domain.Accept(linkerVisitor);
155         soulng::spg::CodeGeneratorVisitor codeGeneratorVisitor(verbose, noParserDebugSupport);
156         domain.Accept(codeGeneratorVisitor);
157     }
158     catch (const std::exception& ex;)
159     {
160         std::cerr << ex.what() << std::endl;
161         return 1;
162     }
163     return 0;
164 }
