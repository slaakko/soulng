 460 void CodeGeneratorVisitor::Visit(ActionParser& parser)
 461 {
 462     if (stage == Stage::generateTokenSwitch)
 463     {
 464         CodeEvaluationVisitor codeEvaluationVisitor;
 465         parser.SuccessCode()->Accept(codeEvaluationVisitor);
 466         bool hasPass = codeEvaluationVisitor.HasPass();
 467         if (hasPass)
 468         {
 469             throw std::runtime_error("token switch does not support pass");
 470         }
 471         if (codeEvaluationVisitor.HasReturn())
 472         {
 473             currentRule->SetHasReturn();
 474         }
 475         stage = Stage::beginGenerateTokenSwitch;
 476         parser.Child()->Accept(*this);
 477         bool nonterminalValue = parser.Child()->IsNonterminal();
 478         bool ptrType = false;
 479         if (currentRule->ReturnType() != nullptr)
 480         {
 481             ptrType = currentRule->ReturnType()->IsPtrType();
 482         }
 483         std::u32string nonterminalName;
 484         soulng::cppcode::TypeId* returnType = nullptr;
 485         if (!ptrType)
 486         {
 487             returnType = currentRule->ReturnType();
 488         }
 489         if (nonterminalValue)
 490         {
 491             NonterminalParser* nt = static_cast<NonterminalParser*>(parser.Child());
 492             nonterminalName = nt->Name();
 493         }
 494         NonterminalCountingVisitor countingVisitor(nonterminalInfos);
 495         parser.SuccessCode()->Accept(countingVisitor);
 496         CodeModifyVisitor modifyVisitor(ptrType, nonterminalName, nonterminalInfos, returnType, noParserDebugSupport, currentRule->Name());
 497         parser.SuccessCode()->Accept(modifyVisitor);
 498         parser.SuccessCode()->Print(*formatter);
 499         stage = Stage::endGenerateTokenSwitch;
 500         parser.Child()->Accept(*this);
 501         stage = Stage::generateTokenSwitch;
 502     }
 503     else
 504     {
 505         int prevSetParentMatchNumber = setParentMatchNumber;
