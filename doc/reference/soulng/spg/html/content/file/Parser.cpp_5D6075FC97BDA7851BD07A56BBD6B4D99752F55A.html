259 ActionParser::ActionParser(soulng::cppcode::CompoundStatement* successCode_, soulng::cppcode::CompoundStatement* failCode_, Parser* child_) :
260     UnaryParser(U"action", child_), successCode(successCode_), failCode(failCode_)
261 {
262     if (successCode && !successCode->IsOwned())
263     {
264         successCode->SetOwned();
265         ownedSuccessCode.reset(successCode);
266     }
267     if (failCode && !failCode->IsOwned())
268     {
269         failCode->SetOwned();
270         ownedFailCode.reset(failCode);
271     }
272 }

289 bool ActionParser::IsActionToken() const
290 {
291     CodeEvaluationVisitor visitor;
292     if (successCode)
293     {
294         successCode->Accept(visitor);
295         if (!visitor.HasReturn()) return false;
296         if (visitor.HasPass()) return false;
297     }
298     return Child()->IsToken();
299 }

333 void NonterminalParser::SetArguments(soulng::cppcode::ExpressionList* args_)
334 {
335     if (args_)
336     {
337         for (const auto& arg : args_->Args())
338         {
339             args.push_back(arg);
340             if (!arg->IsOwned())
341             {
342                 arg->SetOwned();
343                 ownedArgs.push_back(std::unique_ptr<soulng::cppcode::CppObject>(arg));
344             }
345         }
346     }
347 }

416 RuleParser* GrammarParser::GetRule(const std::u32string& ruleName) const
417 {
418     auto it = ruleMap.find(ruleName);
419     if (it != ruleMap.cend())
420     {
421         return it->second;
422     }
423     else
424     {
425         throw std::runtime_error("rule '" + ToUtf8(ruleName) + "' not found from parser '" + ToUtf8(Name()) + "'");
426     }
427 }
