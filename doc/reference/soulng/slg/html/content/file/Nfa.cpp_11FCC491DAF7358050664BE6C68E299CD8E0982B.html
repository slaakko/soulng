 29 std::vector<NfaState*> NfaState::Next(char32_t c) const
 30 {
 31     std::vector<NfaState*> next;
 32     for (const auto& edge : edges)
 33     {
 34         Symbol* symbol = edge.GetSymbol();
 35         if (symbol->Match(c))
 36         {
 37             next.push_back(edge.Next());
 38         }
 39     }
 40     return next;
 41 }

 51 Nfa MakeNfa(LexerContext& lexerContext, Symbol* symbol)
 52 {
 53     NfaState* start = lexerContext.MakeNfaState();
 54     NfaState* end = lexerContext.MakeNfaState();
 55     end->SetAccept(true);
 56     start->AddEdge(NfaEdge(symbol, end));
 57     return Nfa(start, end);
 58 }

 60 Nfa Cat(const Nfa& left, const Nfa& right)
 61 {
 62     Nfa cat(left);
 63     NfaState* leftEnd = cat.End();
 64     leftEnd->SetAccept(false);
 65     NfaState* rightStart = right.Start();
 66     rightStart->SetStatementIndex(-1);
 67     leftEnd->SetEdges(rightStart->Edges());
 68     cat.SetEnd(right.End());
 69     return cat;
 70 }

 72 Nfa Alt(LexerContext& lexerContext, const Nfa& left, const Nfa& right)
 73 {
 74     NfaState* leftStart = left.Start();
 75     NfaState* leftEnd = left.End();
 76     NfaState* rightStart = right.Start();
 77     NfaState* rightEnd = right.End();
 78     NfaState* start = lexerContext.MakeNfaState();
 79     NfaState* end = lexerContext.MakeNfaState();
 80     end->SetAccept(true);
 81     start->AddEdge(NfaEdge(lexerContext.MakeEpsilon(), leftStart));
 82     start->AddEdge(NfaEdge(lexerContext.MakeEpsilon(), rightStart));
 83     Nfa alt;
 84     alt.SetStart(start);
 85     leftEnd->SetAccept(false);
 86     leftEnd->AddEdge(NfaEdge(lexerContext.MakeEpsilon(), end));
 87     rightEnd->SetAccept(false);
 88     rightEnd->AddEdge(NfaEdge(lexerContext.MakeEpsilon(), end));
 89     alt.SetEnd(end);
 90     return alt;
 91 }

 93 Nfa Kleene(LexerContext& lexerContext, const Nfa& nfa)
 94 {
 95     Nfa kleene;
 96     NfaState* start = lexerContext.MakeNfaState();
 97     NfaState* end = lexerContext.MakeNfaState();
 98     end->SetAccept(true);
 99     start->AddEdge(NfaEdge(lexerContext.MakeEpsilon(), end));
100     kleene.SetStart(start);
101     NfaState* nfaStart = nfa.Start();
102     start->AddEdge(NfaEdge(lexerContext.MakeEpsilon(), nfaStart));
103     NfaState* nfaEnd = nfa.End();
104     nfaEnd->SetAccept(false);
105     nfaEnd->AddEdge(NfaEdge(lexerContext.MakeEpsilon(), end));
106     nfaEnd->AddEdge(NfaEdge(lexerContext.MakeEpsilon(), nfaStart));
107     kleene.SetEnd(end);
108     return kleene;
109 }
