 25 void NfaState::SetEdges(const std::std::vector<NfaEdge>&edges_)
 26 {
 27     edges = edges_;
 28     for (NfaEdge& edge : edges)
 29     {

 59 void NfaState::Print(CodeFormatter& formatter)
 60 {
 61     std::string str = std::to_string(id);
 62     if (statementIndex != -1)
 63     {
 64         str.append(" : STATEMENT=").append(std::to_string(statementIndex));
 65     }
 66     if (accept)
 67     {
 68         str.append(" : ACCEPT");
 69     }
 70     if (!prev.empty())
 71     {
 72         str.append(" : prev:");
 73         for (NfaState* p : prev)
 74         {
 75             str.append(" ").append(std::to_string(p->Id()));
 76         }
 77     }
 78     formatter.WriteLine(str);
 79     formatter.IncIndent();
 80     for (NfaEdge& edge : edges)
 81     {
 82         edge.Print(formatter);
 83     }
 84     formatter.DecIndent();
 85 }

 95 Nfa MakeNfa(LexerContext& lexerContext, Symbol* symbol)
 96 {
 97     NfaState* start = lexerContext.MakeNfaState();
 98     NfaState* end = lexerContext.MakeNfaState();
 99     end->SetAccept(true);
100     start->AddEdge(NfaEdge(symbol, end));
101     return Nfa(start, end);
102 }

104 Nfa Cat(const Nfa& left, const Nfa& right)
105 {
106     Nfa cat(left);
107     NfaState* leftEnd = cat.End();
108     leftEnd->SetAccept(false);
109     NfaState* rightStart = right.Start();
110     rightStart->SetStatementIndex(-1);
111     leftEnd->SetEdges(rightStart->Edges());
112     cat.SetEnd(right.End());
113     return cat;
114 }

116 Nfa Alt(LexerContext& lexerContext, const Nfa& left, const Nfa& right)
117 {
118     NfaState* leftStart = left.Start();
119     NfaState* leftEnd = left.End();
120     NfaState* rightStart = right.Start();
121     NfaState* rightEnd = right.End();
122     NfaState* start = lexerContext.MakeNfaState();
123     NfaState* end = lexerContext.MakeNfaState();
124     end->SetAccept(true);
125     start->AddEdge(NfaEdge(lexerContext.MakeEpsilon(), leftStart));
126     start->AddEdge(NfaEdge(lexerContext.MakeEpsilon(), rightStart));
127     Nfa alt;
128     alt.SetStart(start);
129     leftEnd->SetAccept(false);
130     leftEnd->AddEdge(NfaEdge(lexerContext.MakeEpsilon(), end));
131     rightEnd->SetAccept(false);
132     rightEnd->AddEdge(NfaEdge(lexerContext.MakeEpsilon(), end));
133     alt.SetEnd(end);
134     return alt;
135 }
