 19 void PrintUsage()
 20 {
 21     std::cout << "Soul NG Lexer Generator version " << soulng::slg::LexerGeneratorVersionStr() << std::endl;
 22     std::cout << "Usage: slg [options] <file.lexer>" << std::endl;
 23     std::cout << "Options:" << std::endl;
 24     std::cout << "--help | -h" << std::endl;
 25     std::cout << "  Print help and exit." << std::endl;
 26     std::cout << "--verbose | -v" << std::endl;
 27     std::cout << "  Be verbose." << std::endl;
 28     std::cout << "--debug | -d" << std::endl;
 29     std::cout << "  Print character class partition and NFAs to stdout" << std::endl;
 30     std::cout << "--use-ascii-identifier-classes | -a" << std::endl;
 31     std::cout << "  Use ASCII identifier classes." << std::endl;
 32     std::cout << "  By default uses Unicode identifier classes." << std::endl;
 33     std::cout << "--no-identifier-classes | -n" << std::endl;
 34     std::cout << "  No predefined identifier classes" << std::endl;
 35 }
 36 
 37 struct Initializer 
 38 {
 39     Initializer()
 40     {
 41         soulng::util::Init();
 42         soulng::cppcode::Init();
 43     }
 44     ~Initializer()
 45     {
 46         soulng::cppcode::Done();
 47         soulng::util::Done();
 48     }
 49 };
 50 
 51 int main(int argc, const char** argv)
 52 {
 53     Initializer initializer;
 54     try
 55     {
 56         bool verbose = false;
 57         bool debug = false;
 58         soulng::slg::IdentifierClassKind identifierClassKind = soulng::slg::IdentifierClassKind::unicode;
 59         std::string fileName;
 60         for (int i = 1; i < argc; ++i)
 61         {
 62             std::string arg = argv[i];
 63             if (soulng::util::StartsWith(arg, "--"))
 64             {
 65                 if (arg == "--help")
 66                 {
 67                     PrintUsage();
 68                     return 1;
 69                 }
 70                 else if (arg == "--verbose")
 71                 {
 72                     verbose = true;
 73                 }
 74                 else if (arg == "--debug")
 75                 {
 76                     debug = true;
 77                 }
 78                 else if (arg == "--use-ascii-identifier-classes")
 79                 {
 80                     identifierClassKind = soulng::slg::IdentifierClassKind::ascii;
 81                 }
 82                 else if (arg == "--no-identifier-classes")
 83                 {
 84                     identifierClassKind = soulng::slg::IdentifierClassKind::none;
 85                 }
 86                 else
 87                 {
 88                     throw std::runtime_error("unknown option '" + arg + "'");
 89                 }
 90             }
 91             else if (soulng::util::StartsWith(arg, "-"))
 92             {
 93                 std::string options = arg.substr(1);
 94                 if (options.empty())
 95                 {
 96                     throw std::runtime_error("unknown option '-" + arg + "'");
 97                 }
 98                 for (char o : options)
 99                 {
100                     if (o == 'h')
101                     {
102                         PrintUsage();
103                         return 1;
104                     }
105                     else if (o == 'v')
106                     {
107                         verbose = true;
108                     }
109                     else if (o == 'd')
110                     {
111                         debug = true;
112                     }
113                     else if (o == 'a')
114                     {
115                         identifierClassKind = soulng::slg::IdentifierClassKind::ascii;
116                     }
117                     else if (o == 'n')
118                     {
119                         identifierClassKind = soulng::slg::IdentifierClassKind::none;
120                     }
121                     else
122                     {
123                         throw std::runtime_error("unknown option '-" + std::string(1, o) + "'");
124                     }
125                 }
126             }
127             else
128             {
129                 fileName = soulng::util::GetFullPath(arg);
130             }
131         }
132         if (fileName.empty())
133         {
134             PrintUsage();
135             return 1;
136         }
137         if (verbose)
138         {
139             std::cout << "> " << fileName << std::endl;
140         }
141         std::string s = soulng::util::ReadFile(fileName);
142         std::u32string content = soulng::unicode::ToUtf32(s);
143         LexerFileLexer lexer(content, fileName, 0);
144         std::unique_ptr<soulng::slg::LexerFile> lexerFile = LexerFileParser::Parse(lexer);
145         soulng::slg::LexerContext lexerContext(identifierClassKind);
146         lexerContext.SetFileName(fileName);
147         soulng::slg::RegularExpressionParser regularExpressionParser;
148         lexerContext.SetParser(&regularExpressionParser);
149         std::string root = soulng::util::Path::GetDirectoryName(fileName);
150         lexerFile->Process(root, verbose, debug, lexerContext);
151     }
152     catch (const std::exception& ex;)
153     {
154         std::cerr << ex.what() << std::endl;
155         return 1;
156     }
157     return 0;
158 }
