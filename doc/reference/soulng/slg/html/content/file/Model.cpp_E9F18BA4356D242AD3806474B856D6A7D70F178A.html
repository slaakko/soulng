721         }

724     }

741         {
742             statement->Code()->InsertFront(new soulng::cppcode::ExpressionStatement(new soulng::cppcode::InvokeExpr(new soulng::cppcode::IdExpr(U"Retract"))), true);
743         }

745         {
746             Action* action = actions.Get(statement->Action());

748             {
749                 for (const auto& s : action->Code()->Statements())

754             else

757             }
758         }

760         statement->Code()->Print(sourceFormatter);
761     }
762     sourceFormatter.DecIndent();
763     sourceFormatter.WriteLine("}");
764     sourceFormatter.WriteLine("return CONTINUE_TOKEN;");
765     sourceFormatter.DecIndent();
766     sourceFormatter.WriteLine("}");
767     if (verbose)
768     {
769         std::cout << "==> " << lexerSourceFileName << std::endl;
770     }

814 void LexerFile::Process(const std::string& root, bool verbose, bool debug, LexerContext& lexerContext)
815 {
816     for (auto& d : declarations)
817     {
818         d->FillContext(lexerContext);
819     }
820     if (!lexerContext.GetTokens())
821     {
822         throw std::runtime_error("error: no tokens defined");
823     }
824     if (!lexerContext.GetLexer())
825     {
826         throw std::runtime_error("error: no lexer defined");
827     }
828     for (const auto& d : declarations)
829     {
830         d->Process(root, verbose, debug, lexerContext);
831     }
832     lexerContext.MakeCanonicalClasses();
833     lexerContext.MakeClassPartition(debug);
834     lexerContext.MakeClassMap(root, verbose);
835     lexerContext.GetLexer()->MakeMasterNfa(lexerContext, debug);
836     if (debug)
837     {
838         CodeFormatter formatter(std::cout);
839         lexerContext.Print(formatter);
840     }
841     lexerContext.GetLexer()->MakeDfa(lexerContext);
842     if (debug)
843     {
844         CodeFormatter formatter(std::cout);
845         lexerContext.GetLexer()->GetDfa().Print(lexerContext, formatter);
846     }
847     lexerContext.GetLexer()->WriteAutomaton(root, verbose, lexerContext);
848 }
