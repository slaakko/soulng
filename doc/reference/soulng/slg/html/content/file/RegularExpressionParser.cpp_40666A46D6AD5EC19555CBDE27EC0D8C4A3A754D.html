14 Nfa RegularExpressionParser::Parse(const std::u32string& expression, LexerContext* lexerContext, int line)
15 {
16     try
17     {
18         RegExLexer lexer(expression, lexerContext->FileName(), 0);
19         return RegExParser::Parse(lexer, lexerContext);
20     }
21     catch (const std::exception& ex;)
22     {
23         throw std::runtime_error("error parsing regular expression at " + lexerContext->FileName() + ":" + std::to_string(line) + ": " + ex.what());
24     }
25 }
26 
27 } } // namespace soulng::slg
