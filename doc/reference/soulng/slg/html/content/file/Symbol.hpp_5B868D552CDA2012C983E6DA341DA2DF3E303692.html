 23 class Symbol 
 24 {
 25 public:
 26     Symbol();
 27     virtual ~Symbol();
 28     virtual bool Match(char32_t c) = 0;
 29     virtual void Accept(Visitor& visitor) = 0;
 30     virtual bool IsClass() const { return false; }
 31     virtual bool IsChar() const { return false; }
 32     virtual bool IsAny() const { return false; }
 33     virtual bool IsRange() const { return false; }
 34     void SetName(const std::string& name_);
 35     const std::string& Name() const { return name; }
 36     bool Contained() const { return contained; }
 37     void SetContained() { if (!dontSetContained) contained = true; }
 38     void DontSetContained() { dontSetContained = true; }
 39 private:
 40     std::string name;
 41     bool contained;
 42     bool dontSetContained;
 43 };

 66 class Range : public Symbol
 67 {
 68 public:
 69     Range(char32_t start_, char32_t end_);
 70     bool Match(char32_t c) override;
 71     void Accept(Visitor& visitor) override;
 72     bool IsRange() const override { return true; }
 73     bool IsEmpty() const { return start > end; }
 74     char32_t Start() const { return start; }
 75     char32_t End() const { return end; }
 76     void Print(CodeFormatter& formatter);
 77 private:
 78     char32_t start;
 79     char32_t end;
 80 };

 82 inline bool operator==(const Range& left, const Range& right)
 83 {
 84     return left.Start() == right.Start() && left.End() == right.End();
 85 }
 86 
 87 inline bool operator<(const Range& left, const Range& right)
 88 {
 89     if (left.Start() < right.Start()) return true;
 90     if (left.Start() > right.Start()) return false;
 91     return left.End() < right.End();
 92 }
