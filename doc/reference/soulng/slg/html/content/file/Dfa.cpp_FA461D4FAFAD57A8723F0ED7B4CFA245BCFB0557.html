 20 DfaState* DfaState::Next(int i) const
 21 {
 22     if (i >= 0 && i < next.size())
 23     {
 24         return next[i];
 25     }
 26     else
 27     {
 28         return nullptr;
 29     }
 30 }

 32 void DfaState::Print(LexerContext& context, CodeFormatter& formatter)
 33 {
 34     std::string str = std::to_string(id);
 35     str.append("[");
 36     bool first = true;
 37     for (NfaState* nfaState : nfaStates)
 38     {
 39         if (first)
 40         {
 41             first = false;
 42         }
 43         else
 44         {
 45             str.append(", ");
 46         }
 47         str.append(std::to_string(nfaState->Id()));
 48     }
 49     str.append("]");
 50     if (accept)
 51     {
 52         str.append(" : ACCEPT");
 53     }
 54     if (statementIndex != -1)
 55     {
 56         str.append(" : STATEMENT=" + std::to_string(statementIndex));
 57     }
 58     formatter.WriteLine(str);
 59     formatter.IncIndent();
 60     int n = next.size();
 61     for (int i = 0; i < n; ++i)
 62     {
 63         if (next[i])
 64         {
 65             formatter.Write(std::to_string(i));
 66             if (context.Partition()[i]->Ranges().size() == 1)
 67             {
 68                 context.Partition()[i]->Print(formatter);
 69             }
 70             formatter.WriteLine( + " -> " + std::to_string(next[i]->Id()));
 71         }
 72     }
 73     formatter.DecIndent();
 74 }

 81 void Dfa::Finalize()
 82 {
 83     for (const auto& state : states)
 84     {
 85         for (const auto& nfaState : state->NfaStates())
 86         {
 87             if (nfaState->Accept())
 88             {
 89                 state->SetAccept(true);
 90                 if (state->StatementIndex() == -1 || (nfaState->StatementIndex() != -1 && nfaState->StatementIndex() < state->StatementIndex()))
 91                 {
 92                     state->SetStatementIndex(nfaState->StatementIndex());
 93                 }
 94             }
 95         }
 96     }
 97 }

 99 void Dfa::Print(LexerContext& lexerContext, CodeFormatter& formatter)
100 {
101     formatter.WriteLine("DFA:");
102     for (DfaState* state : states)
103     {
104         state->Print(lexerContext, formatter);
105     }
106     formatter.WriteLine();
107 }
