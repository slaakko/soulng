10 DfaState::DfaState(int id_, const std::vector<NfaState*>& nfaStates_) : id(id_), nfaStates(nfaStates_), next(), marked(false), accept(false), statementIndex(-1)

14 void DfaState::AddNext(DfaState* n)
15 {
16     next.push_back(n);
17 }
18 
19 DfaState* DfaState::Next(int i) const
20 {
21     if (i >= 0 && i < next.size())
22     {
23         return next[i];
24     }
25     else
26     {
27         return nullptr;
28     }
29 }

31 void Dfa::AddState(DfaState* state)
32 {
33     states.push_back(state);
34 }
35 
36 void Dfa::Finalize()
37 {
38     for (const auto& state : states)
39     {
40         for (const auto& nfaState : state->NfaStates())
41         {
42             if (nfaState->Accept())
43             {
44                 state->SetAccept(true);
45                 if (state->StatementIndex() == -1 || (nfaState->StatementIndex() != -1 && nfaState->StatementIndex() < state->StatementIndex()))
46                 {
47                     state->SetStatementIndex(nfaState->StatementIndex());
48                 }
49             }
50         }
51     }
52 }
53 
54 } } // namespace soulng::slg
