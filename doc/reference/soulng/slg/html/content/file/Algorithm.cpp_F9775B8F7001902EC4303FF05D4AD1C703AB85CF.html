 12 std::std::vector<NfaState*>EpsilonClosure(conststd::std::vector<NfaState*>&states)
 13 {
 14     std::vector<NfaState*> stack;
 15     for (NfaState* state : states)
 16     {
 17         stack.push_back(state);
 18     }
 19     std::vector<NfaState*> epsilonClosure = states;
 20     while (!stack.empty())
 21     {
 22         NfaState* s = stack.back();
 23         stack.pop_back();
 24         std::vector<NfaState*> u = s->Next(eps);
 25         for (NfaState* v : u)
 26         {
 27             if (std::find(epsilonClosure.cbegin(), epsilonClosure.cend(), v) == epsilonClosure.cend())
 28             {
 29                 epsilonClosure.push_back(v);
 30                 stack.push_back(v);
 31             }
 32         }
 33     }
 34     return epsilonClosure;
 35 }

 43 std::std::vector<NfaState*>Move(conststd::std::vector<NfaState*>&states, char32_tc)
 44 {
 45     std::vector<NfaState*> next;
 46     for (NfaState* state : states)
 47     {
 48         std::vector<NfaState*> n = state->Next(c);
 49         for (NfaState* s : n)
 50         {
 51             if (std::find(next.cbegin(), next.cend(), s) == next.cend())
 52             {
 53                 next.push_back(s);
 54             }
 55         }
 56     }
 57     return next;
 58 }

 60 Dfa Compile(LexerContext& lexerContext, Nfa& nfa)
 61 {
 62     Dfa dfa;
 63     std::vector<NfaState*> start = EpsilonClosure(nfa.Start());
 64     DfaState* s = lexerContext.MakeDfaState(start);
 65     s->Mark();
 66     std::vector<DfaState*> stack;
 67     stack.push_back(s);
 68     while (!stack.empty())
 69     {
 70         DfaState* state = stack.back();
 71         stack.pop_back();
 72         dfa.AddState(state);
 73         for (Class* cls : lexerContext.Partition())
 74         {
 75             if (!cls->Chars().empty())
 76             {
 77                 char32_t c = cls->Chars().front();
 78                 std::vector<NfaState*> next = EpsilonClosure(Move(state->NfaStates(), c));
 79                 if (next.empty())
 80                 {
 81                     state->AddNext(nullptr);
 82                 }
 83                 else
 84                 {
 85                     DfaState* n = lexerContext.MakeDfaState(next);
 86                     state->AddNext(n);
 87                     if (!n->IsMarked())
 88                     {
 89                         n->Mark();
 90                         stack.push_back(n);
 91                     }
 92                 }
 93             }
 94         }
 95     }
 96     dfa.Finalize();
 97     return dfa;
 98 }
