 29 std::vector<NfaState*> NfaState::Next(char32_t c) const
 30 {
 31     std::vector<NfaState*> next;
 32     for (const auto& edge : edges)
 33     {
 34         Symbol* symbol = edge.GetSymbol();
 35         if (symbol->Match(c))
 36         {
 37             next.push_back(edge.Next());
 38         }
 39     }
 40     return next;
 41 }

 51 Nfa MakeNfa(Context& context, Symbol* symbol)
 52 {
 53     NfaState* start = context.MakeNfaState();
 54     NfaState* end = context.MakeNfaState();
 55     end->SetAccept(true);
 56     start->AddEdge(NfaEdge(symbol, end));
 57     return Nfa(start, end);
 58 }

 60 Nfa Cat(const Nfa& left, const Nfa& right)
 61 {
 62     Nfa cat(left);
 63     NfaState* leftEnd = cat.End();
 64     leftEnd->SetAccept(false);
 65     NfaState* rightStart = right.Start();
 66     leftEnd->SetEdges(rightStart->Edges());
 67     cat.SetEnd(right.End());
 68     return cat;
 69 }

 71 Nfa Alt(Context& context, const Nfa& left, const Nfa& right)
 72 {
 73     NfaState* leftStart = left.Start();
 74     NfaState* leftEnd = left.End();
 75     NfaState* rightStart = right.Start();
 76     NfaState* rightEnd = right.End();
 77     NfaState* start = context.MakeNfaState();
 78     NfaState* end = context.MakeNfaState();
 79     end->SetAccept(true);
 80     start->AddEdge(NfaEdge(context.MakeEpsilon(), leftStart));
 81     start->AddEdge(NfaEdge(context.MakeEpsilon(), rightStart));
 82     Nfa alt;
 83     alt.SetStart(start);
 84     leftEnd->SetAccept(false);
 85     leftEnd->AddEdge(NfaEdge(context.MakeEpsilon(), end));
 86     rightEnd->SetAccept(false);
 87     rightEnd->AddEdge(NfaEdge(context.MakeEpsilon(), end));
 88     alt.SetEnd(end);
 89     return alt;
 90 }

 92 Nfa Kleene(Context& context, const Nfa& nfa)
 93 {
 94     Nfa kleene;

 96     NfaState* end = context.MakeNfaState();
 97     end->SetAccept(true);
 98     start->AddEdge(NfaEdge(context.MakeEpsilon(), end));
 99     kleene.SetStart(start);
100     NfaState* nfaStart = nfa.Start();
101     start->AddEdge(NfaEdge(context.MakeEpsilon(), nfaStart));
102     NfaState* nfaEnd = nfa.End();
103     nfaEnd->SetAccept(false);
104     nfaEnd->AddEdge(NfaEdge(context.MakeEpsilon(), end));
105     nfaEnd->AddEdge(NfaEdge(context.MakeEpsilon(), nfaStart));
106     kleene.SetEnd(end);
107     return kleene;
108 }
