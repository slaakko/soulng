41 bool Class::Match(char32_t c)
42 {
43     bool match = false;
44     for (const auto& symbol : symbols)
45     {
46         if (symbol->Match(c))
47         {
48             match = true;
49             break;
50         }
51     }
52     return match != inverse;
53 }
