 13 void ParseHexChar(const std::string& fileName, char32_t& value, const char32_t*& p, const char32_t* e, const soulng::lexer::Token& token)
 14 {
 15     if (p != e)
 16     {
 17         switch (*p)
 18         {
 19         case '0': case '1': case '2': case '3': case '4': case '5': case '6': case '7': case '8': case '9':
 20         {
 21             value = 16 * value + *p - '0';
 22             break;
 23         }
 24         case 'A': case 'B': case 'C': case 'D': case 'E': case 'F':
 25         {
 26             value = 16 * value + 10 + *p - 'A';
 27             break;
 28         }
 29         case 'a': case 'b': case 'c': case 'd': case 'e': case 'f':
 30         {
 31             value = 16 * value + 10 + *p - 'a';
 32             break;
 33         }
 34         }
 35         ++p;
 36     }
 37     else
 38     {
 39         throw std::runtime_error("hex character expected at " + fileName + ":" + std::to_string(token.line) + ": " + ToUtf8(token.match.ToString()));
 40     }
 41 }

 44 char32_t ParseEscape(const std::string& fileName, const char32_t*& p, const char32_t* e, const soulng::lexer::Token& token)
 45 {
 46     char32_t value = '\0';
 47     if (p != e&&(  *p == 'x' || *p == 'X'))
 48     {
 49         ++p;
 50         while (p != e&&(  (*p >= '0' && *p <= '9') || (*p >= 'a' && *p <= 'f') || (*p >= 'A' && *p <= 'F')))
 51         {
 52             ParseHexChar(fileName, value, p, e, token);
 53         }
 54     }
 55     else if (p != e&&(  *p == 'd' || *p == 'D'))
 56     {
 57         ++p;
 58         while (p != e && *p >= '0' && *p <= '9')
 59         {
 60             value = 10 * value + (*p - '0');
 61             ++p;
 62         }
 63     }
 64     else if (p != e&&(  *p >= '0' && *p <= '7'))
 65     {
 66         while (p != e && *p >= '0' && *p <= '7')
 67         {
 68             value = 8 * value + (*p - '0');
 69             ++p;
 70         }
 71     }
 72     else if (p != e && *p == 'u')
 73     {
 74         ++p;
 75         ParseHexChar(fileName, value, p, e, token);
 76         ParseHexChar(fileName, value, p, e, token);
 77         ParseHexChar(fileName, value, p, e, token);
 78         ParseHexChar(fileName, value, p, e, token);
 79     }
 80     else if (p != e && *p == 'U')
 81     {
 82         ++p;
 83         ParseHexChar(fileName, value, p, e, token);
 84         ParseHexChar(fileName, value, p, e, token);
 85         ParseHexChar(fileName, value, p, e, token);
 86         ParseHexChar(fileName, value, p, e, token);
 87         ParseHexChar(fileName, value, p, e, token);
 88         ParseHexChar(fileName, value, p, e, token);
 89         ParseHexChar(fileName, value, p, e, token);
 90         ParseHexChar(fileName, value, p, e, token);
 91     }
 92     else if (p != e)
 93     {
 94         switch (*p)
 95         {
 96         case 'a': value = '\a'; break;
 97         case 'b': value = '\b'; break;
 98         case 'f': value = '\f'; break;
 99         case 'n': value = '\n'; break;
100         case 'r': value = '\r'; break;
101         case 't': value = '\t'; break;
102         case 'v': value = '\v'; break;
103         default: value = *p; break;
104         }
105         ++p;
106     }
107     return value;
108 }

110 char32_t MakeEscapeValue(const std::string& fileName, const soulng::lexer::Token& token)
111 {
112     char32_t escapeValue = '\0';
113     const char32_t* p = token.match.begin;
114     const char32_t* e = token.match.end;
115     if (p != e && *p == '\\')
116     {
117         ++p;
118     }
119     escapeValue = ParseEscape(fileName, p, e, token);
120     if (p != e)
121     {
122         throw std::runtime_error("invalid escape at " + fileName + ":" + std::to_string(token.line) + ": " + ToUtf8(token.match.ToString()));
123     }
124     return escapeValue;
125 }
