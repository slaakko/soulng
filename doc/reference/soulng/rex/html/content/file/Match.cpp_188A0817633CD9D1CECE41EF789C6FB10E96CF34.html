14 std::u32string MakeRegularExpressionPatternFromFilePattern(const std::u32string& filePattern)
15 {
16     std::u32string pattern;
17     for (char32_t c : filePattern)
18     {
19         switch (c)
20         {
21             case '.': pattern.append(U"\\."); break;
22             case '*': pattern.append(U".*"); break;
23             case '?': pattern.append(U"."); break;
24             default: pattern.append(1, c); break;
25         }
26     }
27     return pattern;
28 }
29 
30 bool FilePatternMatch(const std::u32string& filePath, const std::u32string& filePattern)
31 {
32     return PatternMatch(filePath, MakeRegularExpressionPatternFromFilePattern(filePattern));
33 }
34 
35 bool PatternMatch(const std::u32string& str, const std::u32string& regularExpressionPattern)
36 {
37     Context context;
38     Nfa nfa = CompileRegularExpressionPattern(context, regularExpressionPattern);
39     return PatternMatch(str, nfa);
40 }
41 
42 Nfa CompileRegularExpressionPattern(Context& context, const std::u32string& regularExpressionPattern)
43 {
44     RexLexer lexer(regularExpressionPattern, "", 0);
45     return RexParser::Parse(lexer, &context);
46 }
47 
48 Nfa CompileFilePattern(Context& context, const std::u32string& filePattern)
49 {
50     return CompileRegularExpressionPattern(context, MakeRegularExpressionPatternFromFilePattern(filePattern));
51 }
52 
53 bool PatternMatch(const std::u32string& str, Nfa& nfa)
54 {
55     return Match(nfa, str);
56 }
