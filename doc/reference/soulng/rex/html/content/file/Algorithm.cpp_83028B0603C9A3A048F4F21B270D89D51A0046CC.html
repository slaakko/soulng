12 std::vector<NfaState*> EpsilonClosure(const std::vector<NfaState*>& states)
13 {
14     std::vector<NfaState*> stack;
15     for (NfaState* state : states)
16     {
17         stack.push_back(state);
18     }
19     std::vector<NfaState*> epsilonClosure = states;
20     while (!stack.empty())
21     {
22         NfaState* s = stack.back();
23         stack.pop_back();
24         std::vector<NfaState*> u = s->Next(eps);
25         for (NfaState* v : u)
26         {
27             if (std::find(epsilonClosure.cbegin(), epsilonClosure.cend(), v) == epsilonClosure.cend())
28             {
29                 epsilonClosure.push_back(v);
30                 stack.push_back(v);
31             }
32         }
33     }
34     return epsilonClosure;
35 }

37 std::vector<NfaState*> EpsilonClosure(NfaState* state)
38 {
39     std::vector<NfaState*> states(1, state);
40     return EpsilonClosure(states);
41 }

43 std::vector<NfaState*> Move(const std::vector<NfaState*>& states, char32_t c)
44 {
45     std::vector<NfaState*> next;
46     for (NfaState* state : states)
47     {
48         std::vector<NfaState*> n = state->Next(c);
49         for (NfaState* s : n)
50         {
51             if (std::find(next.cbegin(), next.cend(), s) == next.cend())
52             {
53                 next.push_back(s);
54             }
55         }
56     }
57     return next;
58 }

60 bool Match(Nfa& nfa, const std::u32string& s)
61 {
62     std::vector<NfaState*> states = EpsilonClosure(nfa.Start());
63     for (char32_t c : s)
64     {
65         states = EpsilonClosure(Move(states, c));
66     }
67     for (NfaState* state : states)
68     {
69         if (state->Accept()) return true;
70     }
71     return false;
72 }
