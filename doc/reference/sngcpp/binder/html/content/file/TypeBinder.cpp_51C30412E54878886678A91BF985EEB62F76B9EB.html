 24 void TypeBinder::ResolveIdSequence()
 25 {
 26     ClassTypeSymbol* curClass = currentClass;
 27     if (!curClass)
 28     {
 29         if (currentFunction && currentFunction->Parent() && currentFunction->Parent()->IsClassTypeSymbol())
 30         {
 31             curClass = static_cast<ClassTypeSymbol*>(currentFunction->Parent());
 32         }
 33         else if (currentFunctionDeclaration && currentFunctionDeclaration->Parent() && currentFunctionDeclaration->Parent()->IsClassTypeSymbol())
 34         {
 35             curClass = static_cast<ClassTypeSymbol*>(currentFunctionDeclaration->Parent());
 36         }
 37     }
 38     int n = idSequence.size();
 39     bool classResolved = false;
 40     bool functionResolved = false;
 41     for (int i = 0; i < n; ++i)
 42     {
 43         IdentifierNode* id = idSequence[i];
 44         if (n > 1 && !classResolved && curClass && id->Identifier() == curClass->Name())
 45         {
 46             classResolved = true;
 47             symbolTable.MapNode(id, curClass);
 48         }
 49         else if (!functionResolved && currentFunction && id->Identifier() == currentFunction->Name())
 50         {
 51             functionResolved = true;
 52             symbolTable.MapNode(id, currentFunction);
 53         }
 54         else if (!functionResolved && currentFunctionDeclaration && id->Identifier() == currentFunctionDeclaration->Name())
 55         {
 56             functionResolved = true;
 57             symbolTable.MapNode(id, currentFunctionDeclaration);
 58         }
 59     }
 60 }

 62 void TypeBinder::Visit(SourceFileNode& sourceFileNode)
 63 {
 64     Specifier prevAccessSpecifier = currentAccessSpecifier;
 65     currentAccessSpecifier = Specifier::public_;
 66     std::unique_ptr<BoundSourceFile> sourceFile(new BoundSourceFile(sourceFileNode));
 67     sourceFile->AddFileScope();
 68     BoundSourceFile* prevSourceFile = currentSourceFile;
 69     currentSourceFile = sourceFile.get();
 70     containerScope = symbolTable.GlobalNs()->GetContainerScope();
 71     for (const auto& headerFilePath : sourceFileNode.HeaderFilePaths())
 72     {
 73         SourceFileNode* headerFileNode = sourceFileResolver->GetSourceFile(headerFilePath);
 74         if (headerFileNode)
 75         {
 76             for (Node* usingDirectiveOrDeclaration : headerFileNode->UsingDirectivesAndDeclarations())
 77             {
 78                 usingDirectiveOrDeclaration->Accept(*this);
 79             }
 80             if (!headerFileNode->Processed())
 81             {
 82                 headerFileNode->Accept(*this);
 83             }
 84         }
 85     }
 86     if (!sourceFileNode.Processed())
 87     {
 88         fileId = sourceFileNode.Id();
 89         fileName = sourceFileNode.RelativeSourceFilePath();
 90         sourceFileNode.GlobalNs()->Accept(*this);
 91         sourceFileNode.SetProcessed();
 92         sourceFileMap[&sourceFileNode] = currentSourceFile;
 93         sourceFiles.push_back(std::move(sourceFile));
 94     }
 95     currentSourceFile = prevSourceFile;
 96     currentAccessSpecifier = prevAccessSpecifier;
 97 }

318 void TypeBinder::Visit(MemberDeclarationNode& memberDeclarationNode)
319 {
320     FunctionSymbol* prevFunction = currentFunction;
321     FunctionDeclarationSymbol* prevFunctionDeclaration = currentFunctionDeclaration;
322     if (memberDeclarationNode.Declarator())
323     {
324         memberDeclarationNode.Declarator()->Accept(*this);
325         ResolveIdSequence();
326     }
327     if (currentFunction&&(  currentFunction->Specifiers() & Specifier::virtuality) != Specifier::none)
328     {
329         Symbol* parent = currentFunction->Parent();
330         if (parent->IsClassTypeSymbol())
331         {
332             ClassTypeSymbol* cls = static_cast<ClassTypeSymbol*>(parent);
333             cls->AddVirtualFunction(currentFunction);
334             classesHavingVirtualFunctions.insert(cls);
335         }
336     }
337     if (memberDeclarationNode.TypeExpr())
338     {
339         sngcpp::symbols::TypeSymbol* type = ResolveType(symbolTable, containerScope, std::vector<ContainerScope*>(), *currentSourceFile, TypeResolverFlags::none, currentClass, 
340             memberDeclarationNode.TypeExpr());
341         Symbol* symbol = symbolTable.GetSymbolNothrow(&memberDeclarationNode);
342         if (symbol && symbol->IsVariableSymbol())
343         {
344             VariableSymbol* variableSymbol = static_cast<VariableSymbol*>(symbol);
345             variableSymbol->SetAccess(currentAccessSpecifier);
346             variableSymbol->AddSpecifiers(memberDeclarationNode.Specifiers());
347             variableSymbol->SetType(type);
348         }
349         else
350         {
351             if (currentFunction)
352             {
353                 currentFunction->SetReturnType(type);
354             }
355             if (currentFunctionDeclaration)
356             {
357                 currentFunctionDeclaration->SetReturnType(type);
358             }
359         }
360     }
361     currentFunctionDeclaration = prevFunctionDeclaration;
362     currentFunction = prevFunction;
363 }

365 void TypeBinder::Visit(FunctionNode& functionNode)
366 {
367     FunctionSymbol* prevFunction = currentFunction;
368     ContainerScope* prevContainserScope = containerScope;
369     functionNode.Declarator()->Accept(*this);
370     ResolveIdSequence();
371     if (currentFunction && currentFunction->FunctionGroup())
372     {
373         symbolTable.MapNode(&functionNode, currentFunction);
374         FunctionDeclarationSymbol* functionDeclaration = currentFunction->FunctionGroup()->GetFunctionDeclaration(currentFunction->Parameters(), currentFunction->Specifiers());
375         if (functionDeclaration)
376         {
377             currentFunction->AddSpecifiers(functionDeclaration->Specifiers());
378             currentFunction->SetDeclarationSpan(functionDeclaration->GetSpan());
379             currentFunction->SetDeclarationFileId(functionDeclaration->FileId());
380             currentFunction->SetDeclarationFileName(functionDeclaration->FileName());
381             functionDeclaration->SetFunctionDefinition(currentFunction);
382         }
383         if ((currentFunction->Specifiers() & Specifier::virtuality) != Specifier::none)
384         {
385             Symbol* parent = currentFunction->Parent();
386             if (parent->IsClassTypeSymbol())
387             {
388                 ClassTypeSymbol* cls = static_cast<ClassTypeSymbol*>(parent);
389                 cls->AddVirtualFunction(currentFunction);
390                 classesHavingVirtualFunctions.insert(cls);
391             }
392         }
393         containerScope = currentFunction->GetContainerScope();
394         if (functionNode.TypeExpr())
395         {
396             sngcpp::symbols::TypeSymbol* returnType = ResolveType(symbolTable, containerScope, std::vector<ContainerScope*>(), *currentSourceFile, TypeResolverFlags::none, currentClass, 
397                 functionNode.TypeExpr());
398             currentFunction->SetReturnType(returnType);
399         }
400         if (functionNode.Body())
401         {
402             functionNode.Body()->Accept(*this);
403         }
404     }
405     else
406     {
407         if (functionNode.Body())
408         {
409             functionNode.Body()->Accept(*this);
410         }
411     }
412     currentFunction = prevFunction;
413     containerScope = prevContainserScope;
414 }
