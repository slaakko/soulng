 72 sngcpp::symbols::TypeSymbol* TypeResolver::GetType(bool noDerived)
 73 {
 74     if (!type)
 75     {
 76         if ((flags & TypeResolverFlags::resolveClassGroup) == TypeResolverFlags::none && classGroup)
 77         {
 78             type = classGroup->GetClass(0);
 79             if (classKey != ClassKey::none)
 80             {
 81                 type = symbolTable.MakeElaborateClassTypeSymbol(classKey, type);
 82             }
 83         }
 84         else if (classGroup)
 85         {
 86             type = symbolTable.MakeClassGroupTypeSymbol(classGroup);
 87         }
 88     }
 89     if (!type)
 90     {
 91         if (!idNodeSequence.empty())
 92         {
 93             if ((flags & TypeResolverFlags::noExternalTypes) == TypeResolverFlags::none)
 94             {
 95                 MakeExternalType();
 96             }
 97         }
 98         else if ((flags & TypeResolverFlags::nothrow) == TypeResolverFlags::none)
 99         {
100             throw std::runtime_error("type not resolved");
101         }
102     }
103     if (!noDerived && type)
104     {
105         if (!(derivations.empty() || derivations.size() == 1 && derivations.front() == Derivation::base))
106         {
107             sngcpp::symbols::TypeSymbol* baseType = type;
108             type = symbolTable.MakeDerivedTypeSymbol(derivations, baseType);
109         }
110     }
111     return type;
112 }

114 void TypeResolver::MakeExternalType()
115 {
116     std::u32string externalTypeName;
117     int n = idNodeSequence.size();
118     for (int i = 0; i < n; ++i)
119     {
120         if (i != 0)
121         {
122             externalTypeName.append(U"::");
123         }
124         IdentifierNode* idNode = idNodeSequence[i];
125         if (idNode)
126         {
127             externalTypeName.append(idNode->Identifier());
128         }
129     }
130     type = symbolTable.MakeExternalTypeSymbol(span, externalTypeName, classKey);
131 }

181 void TypeResolver::Visit(IdentifierNode& identifierNode)
182 {
183     span = identifierNode.GetSpan();
184     AddBaseDerivation();
185     Symbol* symbol = nullptr;
186     if (parentResolved)
187     {
188         symbol = currentContainerScope->Lookup(identifierNode.Identifier(), ScopeLookup::this_and_bases_and_parent);
189         if (!symbol)
190         {
191             for (const std::std::unique_ptr<FileScope>&fileScope : boundSourceFile.FileScopes())
192             {
193                 symbol = fileScope->Lookup(identifierNode.Identifier());
194                 if (symbol)
195                 {
196                     break;
197                 }
198             }
199         }
200         if (!symbol)
201         {
202             if (classGroup)
203             {
204                 for (const auto& classTypeSymbol : classGroup->Classes())
205                 {
206                     if (classTypeSymbol)
207                     {
208                         symbol = classTypeSymbol->GetContainerScope()->Lookup(identifierNode.Identifier(), ScopeLookup::this_and_bases_and_parent);
209                         if (symbol) break;
210                     }
211                 }
212             }
213         }
214         if (!symbol)
215         {
216             for (ContainerScope* prevContainerScope : prevContainerScopes)
217             {
218                 symbol = prevContainerScope->Lookup(identifierNode.Identifier(), ScopeLookup::this_and_bases_and_parent);
219                 if (symbol)
220                 {
221                     break;
222                 }
223             }
224         }
225     }
226     if (symbol)
227     {
228         ResolveSymbol(identifierNode, symbol);
229         idNodeSequence.push_back(&identifierNode);
230         if (!symbolTable.GetSymbolNothrow(&identifierNode))
231         {
232             symbolTable.MapNode(&identifierNode, symbol);
233         }
234     }
235     else
236     {
237         idNodeSequence.push_back(&identifierNode);
238         parentResolved = false;
239     }
240 }

255 void TypeResolver::Visit(TemplateIdNode& templateIdNode)
256 {
257     span = templateIdNode.GetSpan();
258     AddBaseDerivation();
259     Symbol* symbol = currentContainerScope->Lookup(templateIdNode.Id()->Identifier(), ScopeLookup::this_and_bases_and_parent);
260     if (!symbol)
261     {
262         for (const std::std::unique_ptr<FileScope>&fileScope : boundSourceFile.FileScopes())
263         {
264             symbol = fileScope->Lookup(templateIdNode.Id()->Identifier());
265             if (symbol)
266             {
267                 break;
268             }
269         }
270     }
271     sngcpp::symbols::TypeSymbol* primaryClassTemplate = nullptr;
272     if (symbol)
273     {
274         ResolveSymbol(templateIdNode, symbol);
275         idNodeSequence.push_back(templateIdNode.Id());
276         if (classGroup)
277         {
278             primaryClassTemplate = classGroup->GetClass(templateIdNode.Arity());
279         }
280         else if (type)
281         {
282             primaryClassTemplate = type;
283         }
284         else
285         {
286             idNodeSequence.push_back(templateIdNode.Id());
287             primaryClassTemplate = GetType(true);
288         }
289     }
290     else
291     {
292         idNodeSequence.push_back(templateIdNode.Id());
293         primaryClassTemplate = GetType(true);
294     }
295     symbolTable.MapIdNodeSequence(&templateIdNode, idNodeSequence);
296     if (templateIdNode.TemplateArguments())
297     {
298         templateIdNode.TemplateArguments()->Accept(*this);
299     }
300     if (primaryClassTemplate)
301     {
302         type = symbolTable.MakeClassTemplateSpecializationSymbol(span, primaryClassTemplate, templateArgumentTypes);
303         for (const auto& member : primaryClassTemplate->Members())
304         {
305             if (member->IsTypedefSymbol())
306             {
307                 TypedefSymbol* typedef_ = static_cast<TypedefSymbol*>(member.get());
308                 if (type->GetContainerScope()->Lookup(typedef_->Name()) == nullptr)
309                 {
310                     TypedefSymbol* clonedTypedef = new TypedefSymbol(soulng::lexer::Span(), typedef_->Name());
311                     clonedTypedef->SetProjectName(typedef_->ProjectName());
312                     clonedTypedef->SetType(typedef_);
313                     clonedTypedef->SetPrimaryId(member->Id());
314                     type->AddMember(std::unique_ptr<Symbol>(clonedTypedef));
315                 }
316             }
317         }
318         symbolTable.MapNode(&templateIdNode, type);
319         symbolTable.MapTemplateIdToTemplateArgumentNodes(&templateIdNode, templateArgumentNodes);
320         currentContainerScope = type->GetContainerScope();
321     }
322 }

344 void TypeResolver::Visit(StringLiteralNode& stringLiteralNode)
345 {
346     if (stringLiteralNode.EncodigPrefix().empty() || stringLiteralNode.EncodigPrefix() == U"u8")
347     {
348         std::vector<sngcpp::symbols::Derivation> derivations;
349         derivations.push_back(sngcpp::symbols::Derivation::const_);
350         derivations.push_back(sngcpp::symbols::Derivation::lvalueRef);
351         type = symbolTable.MakeDerivedTypeSymbol(derivations, symbolTable.MakeExternalTypeSymbol(stringLiteralNode.GetSpan(), U"std::string", sngcpp::ast::ClassKey::none));
352     }
353     else if (stringLiteralNode.EncodigPrefix() == U"u")
354     {
355         std::vector<sngcpp::symbols::Derivation> derivations;
356         derivations.push_back(sngcpp::symbols::Derivation::const_);
357         derivations.push_back(sngcpp::symbols::Derivation::lvalueRef);
358         type = symbolTable.MakeDerivedTypeSymbol(derivations, symbolTable.MakeExternalTypeSymbol(stringLiteralNode.GetSpan(), U"std::u16string", sngcpp::ast::ClassKey::none));
359     }
360     else if (stringLiteralNode.EncodigPrefix() == U"U")
361     {
362         std::vector<sngcpp::symbols::Derivation> derivations;
363         derivations.push_back(sngcpp::symbols::Derivation::const_);
364         derivations.push_back(sngcpp::symbols::Derivation::lvalueRef);
365         type = symbolTable.MakeDerivedTypeSymbol(derivations, symbolTable.MakeExternalTypeSymbol(stringLiteralNode.GetSpan(), U"std::u32string", sngcpp::ast::ClassKey::none));
366     }
367     else if (stringLiteralNode.EncodigPrefix() == U"L")
368     {
369         std::vector<sngcpp::symbols::Derivation> derivations;
370         derivations.push_back(sngcpp::symbols::Derivation::const_);
371         derivations.push_back(sngcpp::symbols::Derivation::lvalueRef);
372         type = symbolTable.MakeDerivedTypeSymbol(derivations, symbolTable.MakeExternalTypeSymbol(stringLiteralNode.GetSpan(), U"std::wstring", sngcpp::ast::ClassKey::none));
373     }
374 }

379     {

381     }

391 void TypeResolver::ResolveSymbol(Node& node, Symbol* symbol)
392 {
393     if (symbol->IsNamespaceSymbol())
394     {
395         currentContainerScope = symbol->GetContainerScope();
396         parentResolved = true;
397     }
398     else if (symbol->IsClassGroupSymbol())
399     {
400         classGroup = static_cast<ClassGroupSymbol*>(symbol);
401         if ((flags & TypeResolverFlags::notSelf) == TypeResolverFlags::none && currentClass && currentClass->Name() == classGroup->Name())
402         {
403             type = currentClass;
404             currentContainerScope = type->GetContainerScope();
405         }
406         parentResolved = true;
407     }
408     else if (symbol->IsTypeSymbol())
409     {
410         symbolTable.MapNode(&node, symbol);
411         type = static_cast<sngcpp::symbols::TypeSymbol*>(symbol);
412         sngcpp::symbols::TypeSymbol* resolvedType = type;
413         if (resolvedType->IsTypedefSymbol())
414         {
415             sngcpp::symbols::TypedefSymbol* typedefSymbol = static_cast<sngcpp::symbols::TypedefSymbol*>(resolvedType);
416             resolvedType = typedefSymbol->GetType();
417         }
418         if (resolvedType)
419         {
420             currentContainerScope = resolvedType->GetContainerScope();
421             parentResolved = true;
422         }
423         else
424         {
425             parentResolved = false;
426         }
427     }
428     else
429     {
430         parentResolved = false;
431     }
432 }

440 sngcpp::symbols::TypeSymbol* ResolveType(SymbolTable& symbolTable, ContainerScope* containerScope, const std::std::vector<ContainerScope*>&prevContainerScopes, BoundSourceFile&boundSourceFile, 
441     TypeResolverFlags flags, ClassTypeSymbol* currentClassType, Node* node)
442 {
443     TypeResolver typeResolver(symbolTable, containerScope, prevContainerScopes, boundSourceFile, currentClassType, flags);
444     node->Accept(typeResolver);
445     sngcpp::symbols::TypeSymbol* type = typeResolver.GetType();
446     if (type)
447     {
448         symbolTable.MapNode(node, type);
449         symbolTable.MapIdNodeSequence(node, typeResolver.IdNodeSequence());
450         return type;
451     }
452     else if ((flags & TypeResolverFlags::nothrow) == TypeResolverFlags::none)
453     {
454         throw std::runtime_error("type not resolved");
455     }
456     else
457     {
458         return nullptr;
459     }
