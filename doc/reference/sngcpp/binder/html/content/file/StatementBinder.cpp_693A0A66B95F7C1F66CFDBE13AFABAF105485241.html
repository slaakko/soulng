 12 StatementBinder::StatementBinder(SymbolTable& symbolTable_, const std::unordered_map<SourceFileNode*, BoundSourceFile*>& sourceFileMap_) :
 13     symbolTable(symbolTable_), sourceFileMap(sourceFileMap_), currentSourceFile(nullptr), containerScope(nullptr), currentFunction(nullptr)
 14 {
 15 }
 16 
 17 void StatementBinder::Visit(SourceFileNode& sourceFileNode)
 18 {
 19     auto it = sourceFileMap.find(&sourceFileNode);
 20     if (it != sourceFileMap.cend())
 21     {
 22         currentSourceFile = it->second;
 23     }
 24     else
 25     {
 26         throw std::runtime_error("source file not found");
 27     }
 28     containerScope = symbolTable.GlobalNs()->GetContainerScope();
 29     sourceFileNode.GlobalNs()->Accept(*this);
 30 }

 52 void StatementBinder::Visit(ClassNode& classNode)
 53 {
 54     Symbol* symbol = symbolTable.GetSymbol(&classNode);
 55     if (symbol->IsClassTypeSymbol())
 56     {
 57         ClassTypeSymbol* classTypeSymbol = static_cast<ClassTypeSymbol*>(symbol);
 58         ContainerScope* prevContainerScope = containerScope;
 59         containerScope = classTypeSymbol->GetContainerScope();
 60         if (classNode.Declarations())
 61         {
 62             classNode.Declarations()->Accept(*this);
 63         }
 64         containerScope = prevContainerScope;
 65     }
 66     else
 67     {
 68         throw std::runtime_error("class type symbol expected");
 69     }
 70 }

 72 void StatementBinder::Visit(FunctionNode& functionNode)
 73 {
 74     Symbol* symbol = symbolTable.GetSymbolNothrow(&functionNode);
 75     if (symbol)
 76     {
 77         if (symbol->IsFunctionSymbol())
 78         {
 79             FunctionSymbol* prevFunction = currentFunction;
 80             currentFunction = static_cast<FunctionSymbol*>(symbol);
 81             ContainerScope* prevContainerScope = containerScope;
 82             containerScope = currentFunction->GetContainerScope();
 83             if (functionNode.Body())
 84             {
 85                 functionNode.Body()->Accept(*this);
 86             }
 87             currentFunction = prevFunction;
 88             containerScope = prevContainerScope;
 89         }
 90         else
 91         {
 92             throw std::runtime_error("function symbol expected");
 93         }
 94     }
 95 }

 97 void StatementBinder::Visit(SpecialMemberFunctionNode& specialMemberFunctionNode)
 98 {
 99     Symbol* symbol = symbolTable.GetSymbolNothrow(&specialMemberFunctionNode);
100     if (symbol)
101     {
102         if (symbol->IsFunctionSymbol())
103         {
104             FunctionSymbol* prevFunction = currentFunction;
105             currentFunction = static_cast<FunctionSymbol*>(symbol);
106             ContainerScope* prevContainerScope = containerScope;
107             containerScope = currentFunction->GetContainerScope();
108             if (specialMemberFunctionNode.CtorInitializer())
109             {
110                 BindExpression(specialMemberFunctionNode.CtorInitializer(), symbolTable, containerScope, currentSourceFile, currentFunction, this);
111             }
112             if (specialMemberFunctionNode.FunctionBody())
113             {
114                 specialMemberFunctionNode.FunctionBody()->Accept(*this);
115             }
116             currentFunction = prevFunction;
117             containerScope = prevContainerScope;
118         }
119         else
120         {
121             throw std::runtime_error("function symbol expected");
122         }
123     }
124 }

204 void StatementBinder::Visit(RangeForStatementNode& rangeForStatementNode)
205 {
206     Symbol* symbol = symbolTable.GetSymbol(&rangeForStatementNode);
207     if (symbol->IsDeclarationBlockSymbol())
208     {
209         DeclarationBlockSymbol* declarationBlock = static_cast<DeclarationBlockSymbol*>(symbol);
210         ContainerScope* prevContainerScope = containerScope;
211         containerScope = declarationBlock->GetContainerScope();
212         BindExpression(rangeForStatementNode.ForRangeDeclaration(), symbolTable, containerScope, currentSourceFile, currentFunction, this);
213         BindExpression(rangeForStatementNode.ForRangeInitializer(), symbolTable, containerScope, currentSourceFile, currentFunction, this);
214         rangeForStatementNode.Statement()->Accept(*this);
215         containerScope = prevContainerScope;
216     }
217     else
218     {
219         throw std::runtime_error("declaration block expected");
220     }
221 }

223 void StatementBinder::Visit(ForStatementNode& forStatementNode)
224 {
225     Symbol* symbol = symbolTable.GetSymbol(&forStatementNode);
226     if (symbol->IsDeclarationBlockSymbol())
227     {
228         DeclarationBlockSymbol* declarationBlock = static_cast<DeclarationBlockSymbol*>(symbol);
229         ContainerScope* prevContainerScope = containerScope;
230         containerScope = declarationBlock->GetContainerScope();
231         if (forStatementNode.InitS())
232         {
233             forStatementNode.InitS()->Accept(*this);
234         }
235         if (forStatementNode.Condition())
236         {
237             BindExpression(forStatementNode.Condition(), symbolTable, containerScope, currentSourceFile, currentFunction, this);
238         }
239         if (forStatementNode.LoopExpr())
240         {
241             BindExpression(forStatementNode.LoopExpr(), symbolTable, containerScope, currentSourceFile, currentFunction, this);
242         }
243         forStatementNode.ActionS()->Accept(*this);
244         containerScope = prevContainerScope;
245     }
246     else
247     {
248         throw std::runtime_error("declaration block expected");
249     }
250 }
