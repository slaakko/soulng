289 void ExpressionBinder::Visit(InvokeExpressionNode& invokeExpressionNode)
290 {
291     FunctionGroupSymbol* functionGroup = nullptr;
292     std::vector<Symbol*> subjectSymbols = BindExpression(invokeExpressionNode.Child(), symbolTable, containerScope, boundSourceFile, currentFunction, statementBinder);
293     if (!subjectSymbols.empty())
294     {
295         Symbol* subject = subjectSymbols.front();
296         if (subject)
297         {
298             if (subject->IsFunctionGroupSymbol())
299             {
300                 functionGroup = static_cast<FunctionGroupSymbol*>(subject);
301             }
302             else if (subject->IsPseudoTypeSymbol())
303             {
304                 PseudoTypeSymbol* pseudoTypeSymbol = static_cast<PseudoTypeSymbol*>(subject);
305                 functionGroup = pseudoTypeSymbol->GetFunctionGroup();
306             }
307         }
308     }
309     std::vector<Symbol*> argumentSymbols;
310     if (invokeExpressionNode.Arguments())
311     {
312         argumentSymbols = std::move(BindExpression(invokeExpressionNode.Arguments(), symbolTable, containerScope, boundSourceFile, currentFunction, statementBinder));
313     }
314     if (functionGroup)
315     {
316         FunctionSymbol* functionSymbol = functionGroup->ResolveOverload(argumentSymbols);
317         if (functionSymbol)
318         {
319             symbolTable.MapNode(&invokeExpressionNode, functionSymbol);
320             symbols.push_back(functionSymbol);
321             currentFunction->AddToCalls(functionSymbol);
322             functionSymbol->AddToCalledBy(currentFunction);
323         }
324     }
325 }

421 void ExpressionBinder::Visit(NestedIdNode& nestedIdNode)
422 {
423     std::vector<Symbol*> symbols;
424     if (nestedIdNode.Left())
425     {
426         symbols = std::move(BindExpression(nestedIdNode.Left(), symbolTable, containerScope, boundSourceFile, currentFunction, statementBinder));
427     }
428     else
429     {
430         symbols.push_back(symbolTable.GlobalNs());
431     }
432     Symbol* s = nullptr;
433     if (!symbols.empty())
434     {
435         s = symbols.front();
436     }
437     if (s)
438     {
439         if (s->IsNamespaceSymbol())
440         {
441             containerScope = s->GetContainerScope();
442         }
443         else if (s->IsTypedefSymbol())
444         {
445             TypedefSymbol* typedefSymbol = static_cast<TypedefSymbol*>(s);
446             sngcpp::symbols::TypeSymbol* type = typedefSymbol->GetType();
447             containerScope = type->GetContainerScope();
448         }
449         else if (s->IsTypeSymbol())
450         {
451             containerScope = s->GetContainerScope();
452         }
453         else if (s->IsClassGroupSymbol())
454         {
455             ClassGroupSymbol* group = static_cast<ClassGroupSymbol*>(s);
456             ClassTypeSymbol* cls = group->GetClass(0);
457             if (cls)
458             {
459                 containerScope = cls->GetContainerScope();
460             }
461         }
462     }
463     if (containerScope)
464     {
465         symbols = std::move(BindExpression(nestedIdNode.Right(), symbolTable, containerScope, boundSourceFile, currentFunction, statementBinder));
466     }
467     else
468     {
469         symbols.clear();
470     }
471     this->symbols = std::move(symbols);
472 }

474 void ExpressionBinder::Visit(IdentifierNode& identifierNode)
475 {
476     Symbol* symbol = containerScope->Lookup(identifierNode.Identifier(), ScopeLookup::this_and_bases_and_parent);
477     if (!symbol)
478     {
479         for (const std::unique_ptr<FileScope>& fileScope : boundSourceFile->FileScopes())
480         {
481             symbol = fileScope->Lookup(identifierNode.Identifier());
482             if (symbol)
483             {
484                 break;
485             }
486         }
487     }
488     if (symbol)
489     {
490         symbolTable.MapNode(&identifierNode, symbol);
491         symbols.push_back(symbol);
492     }
493 }

504 std::vector<Symbol*> BindExpression(Node* node, SymbolTable& symbolTable, ContainerScope* containerScope, BoundSourceFile* boundSourceFile, FunctionSymbol* currentFunction, 
505     StatementBinder* statementBinder)
506 {
507     ExpressionBinder expressionBinder(symbolTable, containerScope, boundSourceFile, currentFunction, statementBinder);
508     node->Accept(expressionBinder);
509     std::vector<Symbol*> symbols = expressionBinder.GetSymbols();
510     if (symbols.empty())
