18 enum class TypeResolverFlags : uint8_t 
19 {
20     none=  0,

27 inline TypeResolverFlags operator|(TypeResolverFlags left, TypeResolverFlags right)
28 {
29     return TypeResolverFlags(uint8_t(left) | uint8_t(right));
30 }
31 
32 inline TypeResolverFlags operator&(TypeResolverFlags left, TypeResolverFlags right)
33 {
