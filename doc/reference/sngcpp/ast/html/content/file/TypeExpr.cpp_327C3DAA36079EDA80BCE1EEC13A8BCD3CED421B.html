38 Node* CreatePrefixTypeExprNode(const Span& span_, Specifier cvSpecifiers, Node* typeExprNode)
39 {
40     Node* node = typeExprNode;
41     Specifier specifier = cvSpecifiers & Specifier::volatile_;
42     if (specifier == Specifier::volatile_)
43     {
44         node = new VolatileNode(span_, node);
45     }
46     specifier = cvSpecifiers & Specifier::const_;
47     if (specifier == Specifier::const_)
48     {
49         node = new ConstNode(span_, node);
50     }
51     return node;
52 }
