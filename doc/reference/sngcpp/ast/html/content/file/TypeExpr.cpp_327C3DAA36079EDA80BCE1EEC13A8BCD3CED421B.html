 38 Node* CreatePrefixTypeExprNode(const Span& span_, Specifier cvSpecifiers, Node* typeExprNode)
 39 {
 40     Node* node = typeExprNode;
 41     Specifier specifier = cvSpecifiers & Specifier::volatile_;
 42     if (specifier == Specifier::volatile_)
 43     {
 44         node = new VolatileNode(span_, node);
 45     }
 46     specifier = cvSpecifiers & Specifier::const_;
 47     if (specifier == Specifier::const_)
 48     {
 49         node = new ConstNode(span_, node);
 50     }
 51     return node;
 52 }

106 bool IsConstructorName(Node* node)
107 {
108     if (node->GetNodeType() == NodeType::nestedIdNode)
109     {
110         NestedIdNode* nestedIdNode = static_cast<NestedIdNode*>(node);
111         if (nestedIdNode->Left()->GetNodeType() == NodeType::identifierNode && nestedIdNode->Right()->GetNodeType() == NodeType::identifierNode)
112         {
113             IdentifierNode* left = static_cast<IdentifierNode*>(nestedIdNode->Left());
114             IdentifierNode* right = static_cast<IdentifierNode*>(nestedIdNode->Right());
115             if (left->Identifier() == right->Identifier())
116             {
117                 return true;
118             }
119         }
120     }
121     return false;
122 }
