 13 enum class Suffix : uint8_t 
 14 {
 15     none=  0, 
 16     f=  1 << 0, 
 17     l=  1 << 1, 
 18     ll=  1 << 2, 
 19     u=  1 << 3
 20 };
 21 
 22 inline Suffix operator|(Suffix left, Suffix right)
 23 {
 24     return Suffix(uint8_t(left) | uint8_t(right));
 25 }
 26 
 27 inline Suffix operator&(Suffix left, Suffix right)
 28 {
 29     return Suffix(uint8_t(left) & uint8_t(right));
 30 }
