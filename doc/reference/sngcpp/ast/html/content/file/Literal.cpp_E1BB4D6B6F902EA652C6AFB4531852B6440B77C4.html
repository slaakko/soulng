 13 std::u32string ToString(Suffix suffix)
 14 {
 15     std::u32string s;
 16     if ((suffix & Suffix::u) != Suffix::none)
 17     {
 18         s.append(U"u");
 19     }
 20     if ((suffix & Suffix::f) != Suffix::none)
 21     {
 22         s.append(U"f");
 23     }
 24     if ((suffix & Suffix::l) != Suffix::none)
 25     {
 26         s.append(U"l");
 27     }
 28     if ((suffix & Suffix::ll) != Suffix::none)
 29     {
 30         s.append(U"ll");
 31     }
 32     return s;
 33 }
