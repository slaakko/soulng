 75 void UnaryNode::Write(Writer& writer)
 76 {
 77     Node::Write(writer);
 78     writer.GetBinaryWriter().Write(child != nullptr);
 79     if (child)
 80     {
 81         child->Write(writer);
 82     }
 83 }

 85 void UnaryNode::Read(Reader& reader)
 86 {
 87     Node::Read(reader);
 88     bool hasChild = reader.GetBinaryReader().ReadBool();
 89     if (hasChild)
 90     {
 91         child.reset(reader.ReadNode());
 92     }
 93 }

 95 void UnaryNode::SetFullSpan()
 96 {
 97     if (child)
 98     {
 99         const Span& thisSpan = GetSpan();
100         child->SetFullSpan();
101         const Span& childSpan = child->GetSpan();
102         SetSpanStart(std::min(thisSpan.start, childSpan.start));
103         SetSpanEnd(std::max(thisSpan.end, childSpan.end));
104     }
105 }

129 void BinaryNode::SetFullSpan()
130 {
131     const Span& thisSpan = GetSpan();
132     left->SetFullSpan();
133     right->SetFullSpan();
134     const Span& leftSpan = left->GetSpan();
135     const Span& rightSpan = right->GetSpan();
136     SetSpanStart(std::min(thisSpan.start, std::min(leftSpan.start, rightSpan.start)));
137     SetSpanEnd(std::max(thisSpan.end, std::max(leftSpan.end, rightSpan.end)));
138 }
