44 Project* Solution::GetProject(const std::u32string& guid) const
45 {
46     auto it = projectMap.find(guid);
47     if (it != projectMap.cend())
48     {
49         return it->second;
50     }
51     else
52     {
53         return nullptr;
54     }
55 }

57 void Visit(std::vector<Project*>& buildOrder, Project* project, std::unordered_set<Project*>& visited, std::unordered_set<Project*>& tempVisit, Solution* solution)
58 {
59     if (tempVisit.find(project) == tempVisit.cend())
60     {
61         if (visited.find(project) == visited.cend())
62         {
63             tempVisit.insert(project);
64             for (const std::u32string& dependsOnProjectGuid : project->Dependencies())
65             {
66                 Project* dependsOnProject = solution->GetProject(dependsOnProjectGuid);
67                 Visit(buildOrder, dependsOnProject, visited, tempVisit, solution);
68             }
69             tempVisit.erase(project);
70             visited.insert(project);
71             buildOrder.push_back(project);
72         }
73     }
74 }
75 
76 std::vector<Project*> Solution::BuildOrder()
77 {
78     std::vector<Project*> buildOrder;
79     std::unordered_set<Project*> visited;
80     std::unordered_set<Project*> tempVisit;
81     for (const std::unique_ptr<Project>& project : projects)
82     {
83         if (visited.find(project.get()) == visited.cend())
84         {
85             Visit(buildOrder, project.get(), visited, tempVisit, this);
86         }
87     }
88     return buildOrder;
89 }
