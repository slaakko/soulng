 24 {

 27 SimpleTypeNode::SimpleTypeNode(const Span& span_, const std::std::vector<SimpleTypeSpecifier>&specifiers_):
 28     Node(NodeType::simpleTypeNode, span_), specifiers(specifiers_), flags(SimpleTypeSpecifierFlags::none)
 29 {
 30     SimpleTypeSpecifier prev = SimpleTypeSpecifier::none;
 31     for (auto s : specifiers)
 32     {
 33         switch (s)
 34         {
 35             case SimpleTypeSpecifier::char_: flags = flags | SimpleTypeSpecifierFlags::char_; break;
 36             case SimpleTypeSpecifier::char16: flags = flags | SimpleTypeSpecifierFlags::char16; break;
 37             case SimpleTypeSpecifier::char32: flags = flags | SimpleTypeSpecifierFlags::char32; break;
 38             case SimpleTypeSpecifier::wchar: flags = flags | SimpleTypeSpecifierFlags::wchar; break;
 39             case SimpleTypeSpecifier::bool_: flags = flags | SimpleTypeSpecifierFlags::bool_; break;
 40             case SimpleTypeSpecifier::short_: flags = flags | SimpleTypeSpecifierFlags::short_; break;
 41             case SimpleTypeSpecifier::int_: flags = flags | SimpleTypeSpecifierFlags::int_; break;
 42             case SimpleTypeSpecifier::long_:
 43             {
 44                 if (prev == SimpleTypeSpecifier::long_)
 45                 {
 46                     flags = flags & ~SimpleTypeSpecifierFlags::long_;
 47                     flags = flags | SimpleTypeSpecifierFlags::longLong_;
 48                 }
 49                 else
 50                 {
 51                     flags = flags | SimpleTypeSpecifierFlags::long_;
 52                 }
 53                 break;
 54             }
 55             case SimpleTypeSpecifier::signed_: flags = flags | SimpleTypeSpecifierFlags::signed_; break;
 56             case SimpleTypeSpecifier::unsigned_: flags = flags | SimpleTypeSpecifierFlags::unsigned_; break;
 57             case SimpleTypeSpecifier::float_: flags = flags | SimpleTypeSpecifierFlags::float_; break;
 58             case SimpleTypeSpecifier::double_: flags = flags | SimpleTypeSpecifierFlags::double_; break;
 59             case SimpleTypeSpecifier::void_: flags = flags | SimpleTypeSpecifierFlags::void_; break;
 60             case SimpleTypeSpecifier::auto_: flags = flags | SimpleTypeSpecifierFlags::auto_; break;
 61         }
 62         prev = s;
 63     }
 64 }

 71 void SimpleTypeNode::Write(Writer& writer)
 72 {
 73     Node::Write(writer);
 74     uint8_t n = specifiers.size();
 75     writer.GetBinaryWriter().Write(n);
 76     for (const auto& specifier : specifiers)
 77     {
 78         writer.Write(specifier);
 79     }
 80     writer.GetBinaryWriter().Write(static_cast<uint16_t>(flags));
 81 }
 82 
 83 void SimpleTypeNode::Read(Reader& reader)
 84 {
 85     Node::Read(reader);
 86     uint8_t n = reader.GetBinaryReader().ReadByte();
 87     for (uint8_t i = 0; i < n; ++i)
 88     {
 89         specifiers.push_back(reader.ReadSimpleTypeSpecifier());
 90     }
 91     flags = static_cast<SimpleTypeSpecifierFlags>(reader.GetBinaryReader().ReadUShort());
 92 }

 94 std::u32string SimpleTypeNode::Str()
 95 {
 96     std::u32string str;
 97     bool first = true;
 98     for (SimpleTypeSpecifier specifier : specifiers)
 99     {
100         if (first)
101         {
102             first = false;
103         }
104         else
105         {
106             str.append(U" ");
107         }
108         str.append(ToString(specifier));
109     }
110     return str;
111 }
