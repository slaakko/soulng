13 const char32_t* simpleTypeSpecifierStr[uint8_t(SimpleTypeSpecifier::max)]
14 {
15     U"char", U"char16_t", U"char32_t", U"wchar_t", U"bool", U"short", U"int", U"long", U"signed", U"unsigned", U"float", U"double", U"void", U"auto"

18 std::u32string ToString(SimpleTypeSpecifier specifier)
19 {
20     return simpleTypeSpecifierStr[uint8_t(specifier)];

36 void SimpleTypeNode::Write(Writer& writer)
37 {
38     Node::Write(writer);
39     uint8_t n = specifiers.size();
40     writer.GetBinaryWriter().Write(n);
41     for (const auto& specifier : specifiers)
42     {
43         writer.Write(specifier);
44     }
45 }
46 
47 void SimpleTypeNode::Read(Reader& reader)
48 {
49     Node::Read(reader);
50     uint8_t n = reader.GetBinaryReader().ReadByte();
51     for (uint8_t i = 0; i < n; ++i)
52     {
53         specifiers.push_back(reader.ReadSimpleTypeSpecifier());
54     }
55 }
56 
57 std::u32string SimpleTypeNode::Str()
58 {
59     std::u32string str;
60     bool first = true;
61     for (SimpleTypeSpecifier specifier : specifiers)
62     {
63         if (first)
64         {
65             first = false;
66         }
67         else
68         {
69             str.append(U" ");
70         }
71         str.append(ToString(specifier));
72     }
73     return str;
74 }
