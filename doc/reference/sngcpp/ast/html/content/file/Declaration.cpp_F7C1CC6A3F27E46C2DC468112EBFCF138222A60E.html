173 void LinkageSpecificationNode::Write(Writer& writer)
174 {
175     Node::Write(writer);
176     language->Write(writer);
177     writer.GetBinaryWriter().Write(declarations != nullptr);
178     if (declarations)
179     {
180         declarations->Write(writer);
181     }
182 }
183 
184 void LinkageSpecificationNode::Read(Reader& reader)
185 {
186     Node::Read(reader);
187     language.reset(reader.ReadStringLiteralNode());
188     bool hasDeclarations = reader.GetBinaryReader().ReadBool();
189     if (hasDeclarations)
190     {
191         declarations.reset(reader.ReadNode());
192     }
193 }
194 
195 void LinkageSpecificationNode::AddDeclaration(const Span& span_, Node* declaration)
196 {
197     if (declarations)
198     {
199         declarations.reset(new DeclarationSequenceNode(span_, declarations.release(), declaration));
200     }
201     else
202     {
203         declarations.reset(declaration);
204     }
205 }
