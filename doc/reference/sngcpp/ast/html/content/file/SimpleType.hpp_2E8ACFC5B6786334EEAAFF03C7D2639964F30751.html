15 enum class SimpleTypeSpecifier : uint8_t 
16 {
17     none, char_, char16, char32, wchar, bool_, short_, int_, long_, signed_, unsigned_, float_, double_, void_, auto_, max
18 };
19 
20 enum class SimpleTypeSpecifierFlags : uint16_t 
21 {
22     none=  0, char_=  1 << 0, char16=  1 << 1, char32=  1 << 2, wchar=  1 << 3, bool_=  1 << 4, short_=  1 << 5, int_=  1 << 6, long_=  1 << 7, longLong_=  1 << 8, 
23     signed_=  1 << 9, unsigned_=  1 << 10, float_=  1 << 11, double_=  1 << 12, void_=  1 << 13, auto_=  1 << 14
24 };
25 
26 constexpr, inlineSimpleTypeSpecifierFlags operator|(SimpleTypeSpecifierFlags left, SimpleTypeSpecifierFlags right)
27 {
28     return SimpleTypeSpecifierFlags(uint16_t(left) | uint16_t(right));
29 }
30 
31 constexpr, inlineSimpleTypeSpecifierFlags operator&(SimpleTypeSpecifierFlags left, SimpleTypeSpecifierFlags right)
32 {
33     return SimpleTypeSpecifierFlags(uint16_t(left) & uint16_t(right));
34 }
35 
36 constexpr, inlineSimpleTypeSpecifierFlags operator~(SimpleTypeSpecifierFlags flags)
37 {
38     return SimpleTypeSpecifierFlags(~uint16_t(flags));
39 }
