 6 #include <sngcpp/parser/SourceLineParser.hpp>
 7 #include <sngcpp/lexer/CppTokenLexer.hpp>
 8 #include <sngcpp/lexer/CppTokenLexerTokens.hpp>
 9 
10 void ParseSourceLine(const std::u32string& sourceLine, sngcpp::ast::SourceCodeWriter& sourceCodeWriter)
11 {
12     
13 
14     std::u32string line(sourceLine);
15     line.append(U"\n");
16     CppTokenLexer lexer(line, "", 0);
17     lexer.SetSeparatorChar('\n');
18     lexer.writer = &sourceCodeWriter;
19     ++lexer;
20     int i = 0;
21     while (*lexer != END)
22     {
23         switch (*lexer)
24         {
25             case WS: { soulng::lexer::Token token = lexer.GetToken(i); sourceCodeWriter.WriteSpaces(token.match.ToString()); break; }
26             case LINECOMMENT: { soulng::lexer::Token token = lexer.GetToken(i); sourceCodeWriter.WriteLineComment(token.match.ToString()); return; }
27             case BLOCKCOMMENT: { soulng::lexer::Token token = lexer.GetToken(i); sourceCodeWriter.WriteBlockComment(token.match.ToString()); break; }
28             case KEYWORD: { soulng::lexer::Token token = lexer.GetToken(i); sourceCodeWriter.WriteKeyword(token.match.ToString()); break; }
29             case ID: { soulng::lexer::Token token = lexer.GetToken(i); sourceCodeWriter.WriteIdentifier(token.match.ToString()); break; }
30             case CHARLIT: { soulng::lexer::Token token = lexer.GetToken(i); sourceCodeWriter.WriteChar(token.match.ToString()); break; }
31             case STRINGLIT: { soulng::lexer::Token token = lexer.GetToken(i); sourceCodeWriter.WriteString(token.match.ToString()); break; }
32             case NUMBER: { soulng::lexer::Token token = lexer.GetToken(i); sourceCodeWriter.WriteNumber(token.match.ToString()); break; }
33             case PP: { soulng::lexer::Token token = lexer.GetToken(i); sourceCodeWriter.WritePP(token.match.ToString()); break; }
34             case OTHER: { soulng::lexer::Token token = lexer.GetToken(i); sourceCodeWriter.WriteOther(token.match.ToString()); break; }
35         }
36         ++lexer;
37         ++i;
38     }
39 }
