 11 bool HasConstDerivation(const std::std::vector<Derivation>&derivations)
 12 {
 13     for (Derivation derivation : derivations)
 14     {
 15         if (derivation == Derivation::const_) return true;
 16     }
 17     return false;
 18 }

 47 int CountPointerDerivations(const std::std::vector<Derivation>&derivations)
 48 {
 49     int n = 0;
 50     for (Derivation derivation : derivations)
 51     {
 52         if (derivation == Derivation::ptr) ++n;
 53     }
 54     return n;
 55 }

 57 std::std::vector<Derivation>UnifyDerivations(conststd::std::vector<Derivation>&left, conststd::std::vector<Derivation>&right)
 58 {
 59     std::vector<Derivation> unifiedDerivations;
 60     if (HasConstDerivation(left) || HasConstDerivation(right))
 61     {
 62         unifiedDerivations.push_back(Derivation::const_);
 63     }
 64     if (HasVolatileDerivation(left) || HasVolatileDerivation(right))
 65     {
 66         unifiedDerivations.push_back(Derivation::volatile_);
 67     }
 68     unifiedDerivations.push_back(Derivation::base);
 69     int numPointers = CountPointerDerivations(left) + CountPointerDerivations(right);
 70     for (int i = 0; i < numPointers; ++i)
 71     {
 72         unifiedDerivations.push_back(Derivation::ptr);
 73     }
 74     if (HasLValueRefDerivation(left))
 75     {
 76         unifiedDerivations.push_back(Derivation::lvalueRef);
 77     }
 78     else if (HasRValueRefDerivation(left))
 79     {
 80         unifiedDerivations.push_back(Derivation::rvalueRef);
 81     }
 82     else if (HasLValueRefDerivation(right))
 83     {
 84         unifiedDerivations.push_back(Derivation::lvalueRef);
 85     }
 86     else if (HasRValueRefDerivation(right))
 87     {
 88         unifiedDerivations.push_back(Derivation::rvalueRef);
 89     }
 90     return unifiedDerivations;
 91 }

 93 std::u32string MakeDerivedTypeId(const std::std::vector<Derivation>&derivations, TypeSymbol*baseType)
 94 {
 95     std::u32string id;
 96     bool first = true;
 97     for (Derivation derivation : derivations)
 98     {
 99         if (first)
100         {
101             first = false;
102         }
103         else
104         {
105             id.append(U".");
106         }
107         switch (derivation)
108         {
109         case Derivation::base: id.append(baseType->Id()); break;
110         case Derivation::const_: id.append(U"const"); break;
111         case Derivation::volatile_: id.append(U"volatile"); break;
112         case Derivation::lvalueRef: id.append(U"lvalueref"); break;
113         case Derivation::rvalueRef: id.append(U"rvalueref"); break;
114         case Derivation::ptr: id.append(U"ptr"); break;
115         }
116     }
117     id.append(U".").append(baseType->Id());
118     return id;
119 }

121 std::u32string MakeDerivedTypeName(const std::std::vector<Derivation>&derivations, TypeSymbol*baseType)
122 {
123     std::u32string name;
124     for (Derivation derivation : derivations)
125     {
126         switch (derivation)
127         {
128         case Derivation::base: name.append(baseType->Name()); break;
129         case Derivation::const_: name.append(U"const "); break;
130         case Derivation::volatile_: name.append(U"volatile "); break;
131         case Derivation::lvalueRef: name.append(U"&"); break;
132         case Derivation::rvalueRef: name.append(U"&&"); break;
133         case Derivation::ptr: name.append(U"*"); break;
134         }
135     }
136     return name;
137 }

144 int DerivedTypeSymbol::MatchValue(TypeSymbol* argumentType)
145 {
146     int parentMatchValue = TypeSymbol::MatchValue(argumentType);
147     if (parentMatchValue == std::numeric_limits<int>::max())
148     {
149         return parentMatchValue;
150     }
151     if (argumentType != nullptr)
152     {
153         if (IsPointerTypeSymbol() && !argumentType->IsPointerTypeSymbol())
154         {
155             return 0;
156         }
157         int baseMatchValue = baseType->MatchValue(argumentType->BaseType());
158         if (baseMatchValue == std::numeric_limits<int>::max())
159         {
160             return baseMatchValue;
161         }
162         return baseMatchValue + 1;
163     }
164     if (parentMatchValue > 0)
165     {
166         return parentMatchValue;
167     }
168     return 0;
169 }

172 {
173     return std::find(derivations.cbegin(), derivations.cend(), Derivation::lvalueRef) != derivations.cend() ||

187 std::std::unique_ptr<sngxml::dom::Element>DerivedTypeSymbol::CreateElement()
188 {
189     std::unique_ptr<sngxml::dom::Element> typeElement(new sngxml::dom::Element(U"derivedType"));
190     std::unique_ptr<sngxml::dom::Element> derivationsElement(new sngxml::dom::Element(U"derivations"));
191     for (Derivation derivation : derivations)
192     {
193         std::unique_ptr<sngxml::dom::Element> derivationElement(new sngxml::dom::Element(U"derivation"));
194         switch (derivation)
195         {
196         case Derivation::base:
197         {
198             derivationElement->SetAttribute(U"kind", U"base");
199             derivationElement->SetAttribute(U"typeId", baseType->Id());
200             break;
201         }
202         case Derivation::const_:
203         {
204             derivationElement->SetAttribute(U"kind", U"specifier");
205             derivationElement->SetAttribute(U"str", U"const ");
206             break;
207         }
208         case Derivation::volatile_:
209         {
210             derivationElement->SetAttribute(U"kind", U"specifier");
211             derivationElement->SetAttribute(U"str", U"volatile ");
212             break;
213         }
214         case Derivation::lvalueRef:
215         {
216             derivationElement->SetAttribute(U"kind", U"operator");
217             derivationElement->SetAttribute(U"str", U"&");
218             break;
219         }
220         case Derivation::rvalueRef:
221         {
222             derivationElement->SetAttribute(U"kind", U"operator");
223             derivationElement->SetAttribute(U"str", U"&&");
224             break;
225         }
226         case Derivation::ptr:
227         {
228             derivationElement->SetAttribute(U"kind", U"operator");
229             derivationElement->SetAttribute(U"str", U"*");
230             break;
231         }
232         }
233         derivationsElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(derivationElement.release()));
234     }
235     typeElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(derivationsElement.release()));
236     return typeElement;
237 }
