109 void SymbolCreator::Visit(MemberDeclarationNode& memberDeclarationNode)
110 {
111     specifiers = Specifier::none;
112     bool prevFunctionBodyFollowsOrIsPureDeletedOrDefault = functionBodyFollowsOrIsPureDeletedOrDefault;
113     bool prevWasFunctionDeclaration = wasFunctionDeclaration;
114     bool prevWasArrayDeclaration = wasArrayDeclaration;
115     wasFunctionDeclaration = false;
116     wasArrayDeclaration = false;
117     if (memberDeclarationNode.Declarator())
118     {
119         specifiers = memberDeclarationNode.Specifiers();
120         bool wasPureDeletedOrDefault = false;
121         if ((specifiers&(  Specifier::pure_ | Specifier::delete_ | Specifier::default_)) != Specifier::none)
122         {
123             wasPureDeletedOrDefault = true;
124             functionBodyFollowsOrIsPureDeletedOrDefault = true;
125         }
126         memberDeclarationNode.Declarator()->Accept(*this);
127         if (wasPureDeletedOrDefault)
128         {
129             symbolTable.EndFunction(projectName);
130             symbolTable.EndNameSequence(nameSequenceCount);
131         }
132         if (!wasFunctionDeclaration && !wasArrayDeclaration)
133         {
134             int n = symbolTable.BeginNameSequence(nameSequence);
135             std::u32string variableName = nameSequence.back().first;
136             symbolTable.AddVariable(&memberDeclarationNode, variableName, projectName);
137             symbolTable.EndNameSequence(n);
138         }
139     }
140     wasFunctionDeclaration = prevWasFunctionDeclaration;
141     wasArrayDeclaration = prevWasArrayDeclaration;
142     functionBodyFollowsOrIsPureDeletedOrDefault = prevFunctionBodyFollowsOrIsPureDeletedOrDefault;
143 }

175 void SymbolCreator::Visit(IdentifierNode& identifierNode)
176 {
177     if (processingTemplateArguments)
178     {
179         int n = templateParameters.size();
180         for (int i = 0; i < n; ++i)
181         {
182             TypeSymbol* templateParameterSymbol = templateParameters[i].get();
183             if (templateParameterSymbol->Name() == identifierNode.Identifier())
184             {
185                 templateParameters.erase(templateParameters.begin() + i);
186                 break;
187             }
188         }
189     }
190     else
191     {
192         nameSequence.push_back(std::make_pair(identifierNode.Identifier(), 0));
193         groupName = identifierNode.Identifier();
194         if (groupName == U"ParseXmlContent")
195         {
196             int x = 0;
197         }
198     }
199 }

298 void SymbolCreator::Visit(FunctionDeclaratorNode& functionDeclaratorNode)
299 {
300     wasFunctionDeclaration = true;
301     skip = false;
302     if (functionBodyFollowsOrIsPureDeletedOrDefault)
303     {
304         functionDeclaratorNode.Declarator()->Accept(*this);
305         if (!nameSequence.empty())
306         {
307             if (groupName != U"destructor" && !skip)
308             {
309                 specifiers = specifiers | functionDeclaratorNode.CVSpecifiers();
310                 nameSequenceCount = symbolTable.BeginNameSequence(nameSequence);
311                 symbolTable.BeginFunction(&functionDeclaratorNode, groupName, nameSequence.back().first, templateParameters, specifiers, projectName);
312                 templateParameters.clear();
313                 if (functionDeclaratorNode.Parameters())
314                 {
315                     functionDeclaratorNode.Parameters()->Accept(*this);
316                 }
317             }
318         }
319     }
320     else
321     {
322         functionDeclaratorNode.Declarator()->Accept(*this);
323         if (!nameSequence.empty() && !skip)
324         {
325             specifiers = specifiers | functionDeclaratorNode.CVSpecifiers();
326             int n = symbolTable.BeginNameSequence(nameSequence);
327             symbolTable.BeginFunctionDeclaration(&functionDeclaratorNode, groupName, nameSequence.back().first, specifiers);
328             templateParameters.clear();
329             if (functionDeclaratorNode.Parameters())
330             {
331                 functionDeclaratorNode.Parameters()->Accept(*this);
332             }
333             symbolTable.EndFunctionDeclaration();
334             symbolTable.EndNameSequence(n);
335         }
336     }
337 }

349 void SymbolCreator::Visit(FunctionNode& functionNode)
350 {
351     bool prevFunctionBodyFollowsOrIsPureDeletedOrDefault = functionBodyFollowsOrIsPureDeletedOrDefault;
352     int prevNameSequenceCount = nameSequenceCount;
353     functionBodyFollowsOrIsPureDeletedOrDefault = true;
354     specifiers = functionNode.Specifiers();
355     bool wasDefaultOrDeleted = false;
356     if ((specifiers&(  Specifier::delete_ | Specifier::default_)) != Specifier::none)
357     {
358         wasDefaultOrDeleted = true;
359     }
360     functionNode.Declarator()->Accept(*this);
361     if (functionNode.Body() && groupName != U"destructor")
362     {
363         functionNode.Body()->Accept(*this);
364         symbolTable.EndFunction(projectName);
365         symbolTable.EndNameSequence(nameSequenceCount);
366     }
367     else if (wasDefaultOrDeleted)
368     {
369         symbolTable.EndFunction(projectName);
370         symbolTable.EndNameSequence(nameSequenceCount);
371     }
372     functionBodyFollowsOrIsPureDeletedOrDefault = prevFunctionBodyFollowsOrIsPureDeletedOrDefault;
373     nameSequenceCount = prevNameSequenceCount;
374 }
