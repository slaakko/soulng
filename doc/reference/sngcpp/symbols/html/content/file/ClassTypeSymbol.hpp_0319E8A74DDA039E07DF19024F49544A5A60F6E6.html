35     std::u32string IdStr() override;

37     std::unique_ptr<sngxml::dom::Element> CreateElement() override;
38     void SetFileId(const std::u32string& fileId_) { fileId = fileId_; }

44 private:
45     ClassKey classKey;
46     std::vector<TypeSymbol*> baseClasses;
47     std::vector<TypeSymbol*> derivedClasses;
48     std::vector<std::vector<std::TypeSymbol>>templateParameters;
49     std::u32string fileId;
50     std::string fileName;
51     std::vector<FunctionSymbol*> virtualFunctions;
52     bool isAbstract;
