 19 bool ClassTypeSymbol::HasBaseClass(TypeSymbol* type, int& distance)
 20 {
 21     int n = baseClasses.size();
 22     for (int i = 0; i < n; ++i)
 23     {
 24         TypeSymbol* baseClass = baseClasses[i];
 25         if (baseClass == type)
 26         {
 27             ++distance;
 28             return true;
 29         }
 30         if (baseClass->HasBaseClass(type, distance))
 31         {
 32             ++distance;
 33             return true;
 34         }
 35     }
 36     return false;
 37 }

 39 int ClassTypeSymbol::MatchValue(TypeSymbol* argumentType)
 40 {
 41     int parentMatchValue = TypeSymbol::MatchValue(argumentType);
 42     if (parentMatchValue == std::numeric_limits<int>::max())
 43     {
 44         return parentMatchValue;
 45     }
 46     if (argumentType != nullptr)
 47     {
 48         int distance = 1;
 49         if (argumentType->BaseType()->HasBaseClass(this, distance))
 50         {
 51             return distance;
 52         }
 53     }
 54     if (parentMatchValue > 0)
 55     {
 56         return parentMatchValue;
 57     }
 58     return 0;
 59 }

 98 std::std::unique_ptr<sngxml::dom::Element>ClassTypeSymbol::CreateElement()
 99 {
100     std::unique_ptr<sngxml::dom::Element> classElement(new sngxml::dom::Element(U"class"));
101     classElement->SetAttribute(U"key", sngcpp::ast::ToString(classKey));
102     if (!templateParameters.empty())
103     {
104         std::unique_ptr<sngxml::dom::Element> templateParametersElement(new sngxml::dom::Element(U"templateParameters"));
105         for (const std::std::unique_ptr<TypeSymbol>&templateParameter : templateParameters)
106         {
107             std::unique_ptr<sngxml::dom::Element> templateParameterElement = templateParameter->ToDomElement();
108             templateParametersElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(templateParameterElement.release()));
109         }
110         classElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(templateParametersElement.release()));
111     }
112     if (!baseClasses.empty())
113     {
114         std::unique_ptr<sngxml::dom::Element> baseClassesElement(new sngxml::dom::Element(U"baseClasses"));
115         std::sort(baseClasses.begin(), baseClasses.end(), ByName());
116         for (TypeSymbol* baseClass : baseClasses)
117         {
118             std::unique_ptr<sngxml::dom::Element> baseClassElement(new sngxml::dom::Element(U"baseClass"));
119             baseClassElement->SetAttribute(U"baseClassId", baseClass->Id());
120             baseClassesElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(baseClassElement.release()));
121         }
122         classElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(baseClassesElement.release()));
123     }
124     if (!derivedClasses.empty())
125     {
126         std::unique_ptr<sngxml::dom::Element> derivedClassesElement(new sngxml::dom::Element(U"derivedClasses"));
127         std::sort(derivedClasses.begin(), derivedClasses.end(), ByName());
128         for (TypeSymbol* derivedClass : derivedClasses)
129         {
130             std::unique_ptr<sngxml::dom::Element> derivedClassElement(new sngxml::dom::Element(U"derivedClass"));
131             derivedClassElement->SetAttribute(U"derivedClassId", derivedClass->Id());
132             derivedClassesElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(derivedClassElement.release()));
133         }
134         classElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(derivedClassesElement.release()));
135     }
136     if (GetSpan().Valid())
137     {
138         classElement->SetAttribute(U"definitionFile", ToUtf32(fileName));
139         classElement->SetAttribute(U"definitionLine", ToUtf32(std::to_string(GetSpan().line)));
140         classElement->SetAttribute(U"definitionFileId", fileId);
141     }
142     if (isAbstract)
143     {
144         classElement->SetAttribute(U"abstract", U"true");
145     }
146     return classElement;
147 }
