  77 std::u32string HtmlSourceCodeGenerator::MakeSymbolRef(Symbol* symbol)
  78 {
  79     if (!symbol) return std::u32string();
  80     if (symbol->IsParentSymbol())
  81     {
  82         if (symbol->IsClassTemplateSpecializationSymbol())
  83         {
  84             ClassTemplateSpecializationSymbol* specialization = static_cast<ClassTemplateSpecializationSymbol*>(symbol);
  85             return MakeSymbolRef(specialization->PrimaryClassTemplateSymbol());
  86         }
  87         else if (symbol->IsNamespaceSymbol())
  88         {
  89             if (symbol->ProjectName() == projectName)
  90             {
  91                 return ToUtf32(Path::Combine(contentDirPath, ToUtf8(symbol->Id()) + ".html"));
  92             }
  93             else
  94             {
  95                 NamespaceSymbol* ns = static_cast<NamespaceSymbol*>(symbol);

  97                 {
  98                     return ToUtf32(Path::Combine(contentDirPath, ToUtf8(symbol->Id()) + ".html"));
  99                 }
 100                 else
 101                 {
 102                     std::u32string project = ns->FirstProject();
 103                     if (project.empty()) return std::u32string();
 104                     return ToUtf32(contentFilePathResolver->ResolveContentFilePath(projectName, project, contentDirPath, symbol->Id()));
 105                 }
 106             }
 107         }
 108         else
 109         {
 110             return ToUtf32(contentFilePathResolver->ResolveContentFilePath(projectName, symbol->ProjectName(), contentDirPath, symbol->Id()));
 111         }
 112     }
 113     else if (symbol->IsFunctionGroupSymbol() || symbol->IsClassGroupSymbol() || !symbol->Parent() ||

 115     {
 116         return std::u32string();

 171 void HtmlSourceCodeGenerator::WriteId(IdentifierNode* id, Symbol* symbol, bool writeType)
 172 {
 173     MoveTo(id->GetSpan());
 174     if (symbol)
 175     {
 176         if (symbol->IsClassGroupSymbol())
 177         {
 178             ClassGroupSymbol* group = static_cast<ClassGroupSymbol*>(symbol);
 179             symbol = group->GetClass(0);
 180         }
 181         else if (symbol->IsFunctionDeclarationSymbol())
 182         {
 183             FunctionDeclarationSymbol* functionDeclarationSymbol = static_cast<FunctionDeclarationSymbol*>(symbol);
 184             if (functionDeclarationSymbol->FunctionDefinition())
 185             {
 186                 symbol = functionDeclarationSymbol->FunctionDefinition();
 187             }
 188             else
 189             {
 190                 symbol = nullptr;
 191             }
 192         }
 193         writer.WriteLink(id->Identifier(), MakeSymbolRef(symbol));
 194         writer.WriteInlineLink(id->Identifier(), MakeInlineSymbolRef(symbol));
 195     }
 196     else if (writeType)
 197     {
 198         writer.WriteType(id->Identifier());
 199     }
 200     else
 201     {
 202         if (id->IsDtorIdNode())
 203         {
 204             writer.WriteOther(U"~");
 205             writer.WriteIdentifier(id->Identifier());
 206         }
 207         else if (id->IsOperatorFunctionIdNode())
 208         {
 209             OperatorFunctionIdNode* operatorFunctionIdNode = static_cast<OperatorFunctionIdNode*>(id);
 210             writer.WriteKeyword(U"operator");
 211             writer.WriteOther(operatorFunctionIdNode->OpStr());
 212         }
 213         else if (id->IsConversionFunctionIdNode())
 214         {
 215             writer.WriteKeyword(U"operator");
 216             writer.WriteSpace(1);
 217         }
 218         else
 219         {
 220             writer.WriteIdentifier(id->Identifier());
 221         }
 222     }

 249 void HtmlSourceCodeGenerator::WriteType(TypeSymbol* type, const std::vector<IdentifierNode*>& idNodeSequence, TypeExprNode* typeExprNode)
 250 {
 251     int n = idNodeSequence.size();
 252     if (type->IsExternalTypeSymbol())
 253     {
 254         ExternalTypeSymbol* externalTypeSymbol = static_cast<ExternalTypeSymbol*>(type);
 255         if (externalTypeSymbol->GetClassKey() != ClassKey::none)
 256         {
 257             writer.WriteKeyword(ToString(externalTypeSymbol->GetClassKey()));
 258             writer.WriteSpace(1);
 259         }
 260         for (int i = 0; i < n - 1; ++i)
 261         {
 262             IdentifierNode* idNode = idNodeSequence[i];
 263             if (idNode)
 264             {
 265                 writer.WriteIdentifier(idNode->Identifier());
 266             }
 267             writer.WriteOther(U"::");
 268         }
 269         if (n > 0)
 270         {
 271             writer.WriteType(idNodeSequence.back()->Identifier());
 272         }
 273         else
 274         {
 275             writer.WriteType(type->Name());
 276         }
 277     }
 278     else if (type->IsSimpleTypeSymbol())
 279     {
 280         SimpleTypeSymbol* simpleTypeSymbol = static_cast<SimpleTypeSymbol*>(type);
 281         std::vector<std::u32string> components = Split(simpleTypeSymbol->Name(), ' ');
 282         bool first = true;
 283         for (const std::u32string& component : components)
 284         {
 285             if (first)
 286             {
 287                 first = false;
 288             }
 289             else
 290             {
 291                 writer.WriteSpace(1);
 292             }
 293             writer.WriteKeyword(component);
 294         }
 295     }
 296     else if (type->IsDerivedTypeSymbol())
 297     {
 298         DerivedTypeSymbol* derivedTypeSymbol = static_cast<DerivedTypeSymbol*>(type);
 299         for (Derivation derivation : derivedTypeSymbol->Derivations())
 300         {
 301             switch (derivation)
 302             {
 303             case Derivation::const_:
 304             {
 305                 writer.WriteKeyword(U"const ");
 306                 break;
 307             }
 308             case Derivation::volatile_:
 309             {
 310                 writer.WriteKeyword(U"volatile ");
 311                 break;
 312             }
 313             case Derivation::ptr:
 314             {
 315                 writer.WriteOther(U"*");
 316                 break;
 317             }
 318             case Derivation::lvalueRef:
 319             {
 320                 writer.WriteOther(U"&");
 321                 break;
 322             }
 323             case Derivation::rvalueRef:
 324             {
 325                 writer.WriteOther(U"&&");
 326                 break;
 327             }
 328             case Derivation::base:
 329             {
 330                 TypeSymbol* baseType = derivedTypeSymbol->BaseType();
 331                 WriteType(baseType, idNodeSequence, typeExprNode);
 332                 break;
 333             }
 334             }
 335         }
 336     }
 337     else if (type->IsClassTemplateSpecializationSymbol())
 338     {
 339         ClassTemplateSpecializationSymbol* classTemplateSpecializationSymbol = static_cast<ClassTemplateSpecializationSymbol*>(type);
 340         TypeSymbol* primaryType = classTemplateSpecializationSymbol->PrimaryClassTemplateSymbol();
 341         TemplateIdNode* templateIdNode = classTemplateSpecializationSymbol->GetTemplateIdNode();
 342         if (typeExprNode)
 343         {
 344             TemplateIdNode* tinode = symbolTable.GetTemplateIdNodeForTypeExprNode(typeExprNode);
 345             if (tinode)
 346             {
 347                 templateIdNode = tinode;
 348             }
 349         }
 350         WriteType(primaryType, symbolTable.GetIdNodeSequence(templateIdNode), typeExprNode);
 351         writer.WriteOther(U"<");
 352         int n = classTemplateSpecializationSymbol->TemplateArgumentSymbols().size();
 353         for (int i = 0; i < n; ++i)
 354         {
 355             if (i > 0)
 356             {
 357                 writer.WriteOther(U", ");
 358             }
 359             TypeSymbol* templateArgumentSymbol = classTemplateSpecializationSymbol->TemplateArgumentSymbols()[i];
 360             Node* templateArgumentNode = classTemplateSpecializationSymbol->TemplateArgumentNodes()[i];
 361             Node* tanode = symbolTable.GetTemplateArgumentNodeForTypeExprNode(typeExprNode);
 362             if (tanode)
 363             {
 364                 templateArgumentNode = tanode;
 365             }
 366             WriteType(templateArgumentSymbol, symbolTable.GetIdNodeSequence(templateArgumentNode), typeExprNode);
 367         }
 368         writer.WriteOther(U">");
 369     }
 370     else if (type->IsElaborateClassTypeSymbol())
 371     {
 372         ElaborateClassTypeSymbol* elaborateClassTypeSymbol = static_cast<ElaborateClassTypeSymbol*>(type);
 373         writer.WriteKeyword(ToString(elaborateClassTypeSymbol->GetClassKey()));
 374         writer.WriteSpace(1);
 375         WriteType(elaborateClassTypeSymbol->GetClassType(), idNodeSequence, typeExprNode);
 376     }
 377     else
 378     {
 379         for (int i = 0; i < n - 1; ++i)
 380         {
 381             IdentifierNode* idNode = idNodeSequence[i];
 382             writer.WriteIdentifier(idNode->Identifier());
 383             writer.WriteOther(U"::");
 384         }
 385         writer.WriteLink(type->Name(), MakeSymbolRef(type));
 386         writer.WriteInlineLink(type->Name(), MakeInlineSymbolRef(type));
 387     }

 391 {
 392     int lineNumber = span.line;
 393     while (lineNumber > currentSourceLineNumber)
 394     {

 399         else

 402         }
 403     }

 406         OpenLine();
 407     }

 417     lineElement.reset(new sngxml::dom::Element(U"span"));
 418     writer.SetLineElement(lineElement.get());

 423     writer.WriteLineNumberKind(lineNumberText, WriteKind::line);

 426     {
 427         throw std::runtime_error("invalid line starts vector");
 428     }

 430     writer.SetCurrentPos(lineStart);
