  78 void Project::ReadChildren()
  79 {
  80     if (childrenRead) return;
  81     childrenRead = true;
  82     std::unique_ptr<sngxml::xpath::XPathObject> result = sngxml::xpath::Evaluate(U"/gendoc/child", doc.get());
  83     if (result)
  84     {
  85         if (result->Type() == sngxml::xpath::XPathObjectType::nodeSet)
  86         {
  87             sngxml::xpath::XPathNodeSet* nodeSet = static_cast<sngxml::xpath::XPathNodeSet*>(result.get());
  88             for (int i = 0; i < nodeSet->Length(); ++i)
  89             {
  90                 sngxml::dom::Node* node = (*nodeSet)[i];
  91                 if (node->GetNodeType() == sngxml::dom::NodeType::elementNode)
  92                 {
  93                     sngxml::dom::Element* element = static_cast<sngxml::dom::Element*>(node);
  94                     std::u32string project = element->GetAttribute(U"project");
  95                     if (!project.empty())
  96                     {
  97                         std::string childFilePath = Path::Combine(Path::Combine(rootDir, ToUtf8(project)), "gendoc.xml");
  98                         children.push_back(childFilePath);
  99                         std::u32string childProject = project;
 100                         std::unique_ptr<sngxml::dom::Document> childDoc = sngxml::dom::ReadDocument(childFilePath);
 101                         std::unique_ptr<sngxml::xpath::XPathObject> childResult = sngxml::xpath::Evaluate(U"/gendoc/project", childDoc.get());
 102                         if (childResult)
 103                         {
 104                             if (childResult->Type() == sngxml::xpath::XPathObjectType::nodeSet)
 105                             {
 106                                 sngxml::xpath::XPathNodeSet* childNodeSet = static_cast<sngxml::xpath::XPathNodeSet*>(childResult.get());
 107                                 for (int i = 0; i < childNodeSet->Length(); ++i)
 108                                 {

 111                                     {

 117                                             childProjects.push_back(std::move(childProject));
 118                                             childNames.push_back(childName);

 120                                         }
 121                                     }
 122                                 }

 132 void Project::RunChildren(bool verbose, bool rebuild, bool clean)
 133 {
 134     ReadChildren();
 135     if (children.empty()) return;
 136     if (clean)
 137     {
 138         if (verbose)
 139         {
 140             std::cout << ToUtf8(name) << "> cleaning " << children.size() << " child projects..." << std::endl;
 141         }
 142         RunChildrenWithFlags(verbose, false, "l");
 143         if (verbose)
 144         {
 145             std::cout << ToUtf8(name) << "> " << children.size() << " child projects cleaned." << std::endl;
 146         }
 147     }
 148     else
 149     {
 150         if (verbose)
 151         {
 152             std::cout << ToUtf8(name) << "> building " << children.size() << " child projects..." << std::endl;
 153             for (const auto& child : children)
 154             {
 155                 boost::filesystem::remove(GetFullPath(Path::Combine(Path::GetDirectoryName(child), "status.msg")));
 156             }
 157             std::cout << ToUtf8(name) << "> generating ASTs..." << std::endl;
 158         }
 159         RunChildrenWithFlags(verbose, rebuild, "a");
 160         if (verbose)
 161         {
 162             std::cout << ToUtf8(name) << "> ASTs ready." << std::endl;
 163             std::cout << ToUtf8(name) << "> generating content..." << std::endl;
 164         }
 165         RunChildrenWithFlags(verbose, rebuild, "ce");
 166         if (verbose)
 167         {
 168             std::cout << ToUtf8(name) << "> content ready." << std::endl;
 169             for (const auto& child : children)
 170             {
 171                 std::string statusMessageFilePath = GetFullPath(Path::Combine(Path::GetDirectoryName(child), "status.msg"));
 172                 if (boost::filesystem::exists(statusMessageFilePath))
 173                 {
 174                     std::string statusMessage = soulng::util::ReadFile(statusMessageFilePath);
 175                     std::cout << statusMessage;
 176                 }
 177             }
 178             std::cout << ToUtf8(name) << "> " << children.size() << " child projects built or up-to-date." << std::endl;
 179         }
 180     }

 183 void Project::RunChildrenWithFlags(bool verbose, bool rebuild, const std::string& flags)
 184 {
 185     std::string f = "-";
 186     if (verbose)
 187     {
 188         f.append("v");
 189     }
 190     if (rebuild)
 191     {
 192         f.append("r");
 193     }
 194     f.append(flags);
 195     std::vector<boost::process::ipstream> is;
 196     std::vector<boost::process::ipstream> es;
 197     std::vector<boost::process::child> childProcesses;
 198     for (int i = 0; i < children.size(); ++i)
 199     {
 200         if (verbose)
 201         {
 202             std::cout << ToUtf8(name) + ">> " << children[i] << std::endl;
 203         }
 204         is.push_back(boost::process::ipstream());
 205         es.push_back(boost::process::ipstream());
 206         boost::process::child child = boost::process::child(GetPathToExecutable(), f, children[i], boost::process::std_out > is[i], boost::process::std_err > es[i]);
 207         childProcesses.push_back(std::move(child));
 208     }
 209     bool running = true;
 210     while (running)
 211     {
 212         running = false;
 213         for (int i = 0; i < childProcesses.size(); ++i)
 214         {
 215             if (childProcesses[i].running())
 216             {
 217                 std::string line;
 218                 if (std::getline(is[i], line) && !line.empty())
 219                 {
 220                     std::cout << line << std::endl;
 221                 }
 222                 running = true;
 223             }
 224         }
 225     }
 226     for (int i = 0; i < childProcesses.size(); ++i)
 227     {
 228         childProcesses[i].wait();
 229         int result = childProcesses[i].exit_code();
 230         if (result != 0)
 231         {
 232             std::string line;
 233             while (std::getline(es[i], line) && !line.empty())
 234             {
 235                 std::cerr << line << std::endl;
 236             }
 237             throw std::runtime_error("child returned exit code " + std::to_string(result));
 238         }
 239     }

 740 bool Project::BuildAst(bool verbose, bool rebuild)
 741 {
 742     if (!rebuild && boost::filesystem::exists(astFilePath))
 743     {
 744         bool upToDate = true;
 745         for (const File& file : filePaths)
 746         {
 747             if (boost::filesystem::last_write_time(file.Path()) >= boost::filesystem::last_write_time(astFilePath))
 748             {
 749                 upToDate = false;
 750                 break;
 751             }
 752         }
 753         if (upToDate)
 754         {
 755             return false;
 756         }
 757     }
 758     sngcpp::pp::EvaluationContext evaluationContext;
 759     int fileIndex = 0;
 760     for (const File& file : filePaths)
 761     {
 762         if (file.Included())
 763         {
 764             std::unique_ptr<sngcpp::ast::SourceFileNode> sourceFile;
 765             std::unique_ptr<CppLexer> lexer;
 766             if (verbose)
 767             {
 768                 std::cout << ToUtf8(name) + "> " << file.Path() << std::endl;
 769             }
 770             std::string ext = Path::GetExtension(file.Path());
 771             if (ext == ".parser" || ext == ".lexer")
 772             {
 773                 std::unique_ptr<std::u32string> content(new std::u32string(ToUtf32(soulng::util::ReadFile(file.Path()))));
 774                 sourceFile.reset(new sngcpp::ast::SourceFileNode(soulng::lexer::Span(), file.Path(), ToUtf8(file.Name()), name));
 775                 sourceFile->SetSourceFileIndex(fileIndex);
 776                 sourceFile->SetText(std::move(*content));
 777             }
 778             else
 779             {
 780                 sngcpp::pp::PP pp(evaluationContext);
 781                 pp.root = projectRoot;
 782                 pp.includePath = Split(includePath, ';');
 783                 pp.projectHeaderFileSet = &headerFiles;
 784                 pp.Define(sngcpp::pp::ndebug);
 785                 if (verbose)
 786                 {
 787                     pp.verbose = true;
 788                 }
 789                 Preprocess(file.Path(), &pp);
 790                 std::unique_ptr<std::u32string> content(new std::u32string(std::move(pp.ctext)));
 791                 lexer.reset(new CppLexer(content->c_str(), content->c_str() + content->length(), file.Path(), fileIndex));
 792                 lexer->SetSeparatorChar('\n');
 793                 sourceFile.reset(new sngcpp::ast::SourceFileNode(lexer->GetSpan(), file.Path(), ToUtf8(file.Name()), name));
 794                 sourceFile->SetSourceFileIndex(fileIndex);
 795                 SourceFileParser::Parse(*lexer, sourceFile.get());
 796                 sourceFile->SetHeaderFilePaths(std::move(pp.headerFilePaths));
 797                 sourceFile->SetText(std::move(*content));
 798                 sourceFile->ComputeLineStarts();
 799             }
 800             if (ast)
 801             {
 802                 soulng::lexer::Span span = ast->GetSpan();
 803                 ast.reset(new sngcpp::ast::SourceFileSequenceNode(span, ast.release(), sourceFile.release()));
 804             }
 805             else
 806             {
 807                 ast.reset(sourceFile.release());
 808             }
 809             ++fileIndex;
 810             lexers.push_back(std::unique_ptr<soulng::lexer::Lexer>(lexer.release()));
 811         }
 812     }
 813     return true;
 814 }

1003 void Project::ReadDocumentationXml(bool verbose, bool& rebuild)
1004 {
1005     std::unique_ptr<sngxml::xpath::XPathObject> result = sngxml::xpath::Evaluate(U"/gendoc/project/documentation", doc.get());
1006     if (result)
1007     {
1008         if (result->Type() == sngxml::xpath::XPathObjectType::nodeSet)
1009         {
1010             sngxml::xpath::XPathNodeSet* nodeSet = static_cast<sngxml::xpath::XPathNodeSet*>(result.get());
1011             int n = nodeSet->Length();
1012             for (int i = 0; i < n; ++i)
1013             {
1014                 sngxml::dom::Node* node = (*nodeSet)[i];
1015                 if (node->GetNodeType() == sngxml::dom::NodeType::elementNode)
1016                 {
1017                     sngxml::dom::Element* element = static_cast<sngxml::dom::Element*>(node);
1018                     std::u32string documentationFileAttribute = element->GetAttribute(U"file");
1019                     if (!documentationFileAttribute.empty())
1020                     {
1021                         documentationXmlFileName = GetFullPath(Path::Combine(rootDir, ToUtf8(documentationFileAttribute)));
1022                         if (boost::filesystem::exists(documentationXmlFileName))
1023                         {
1024                             documentationXml = sngxml::dom::ReadDocument(documentationXmlFileName);
1025                             if (boost::filesystem::exists(contentXmlFilePath))
1026                             {
1027                                 if (boost::filesystem::last_write_time(documentationXmlFileName) > boost::filesystem::last_write_time(contentXmlFilePath))
1028                                 {
1029                                     rebuild = true;
1030                                 }
1031                             }
1032                         }
1033                         else
1034                         {
1035                             std::cerr << "Documentation XML file '" + documentationXmlFileName + "' does not exist." << std::endl;
1036                         }
1037                     }
1038                 }
1039             }
1040         }
1041     }

1103         }
1104         else
